1//! Graph database schema definition.
2//!
3//! This module defines the vertex types, edge types, and property schemas
4//! used in the Erdős Graph database.
5//!
6//! # Graph Structure
7//! - **Person vertices**: Represent authors with name, Erdős number, and aliases
8//! - **Publication vertices**: Represent papers with title, year, venue
9//! - **AUTHORED edges**: Connect Person vertices to their publications
10//! - **COAUTHORED_WITH edges**: Connect co-authors with weights representing collaboration count
11
12use std::collections::HashMap;
13
14/// Vertex type identifier for Person nodes
15pub const PERSON_TYPE: &str = "Person";
16
17/// Vertex type identifier for Publication nodes
18pub const PUBLICATION_TYPE: &str = "Publication";
19
20/// Edge type for authorship relationships (Person -> Publication)
21pub const AUTHORED_TYPE: &str = "AUTHORED";
22
23/// Edge type for co-authorship relationships (Person -> Person)
24pub const COAUTHORED_WITH_TYPE: &str = "COAUTHORED_WITH";
25
26/// Returns the property schema for Person vertices.
27///
28/// # Properties
29/// - `name`: The person's canonical name
30/// - `erdos_number`: String representation of Erdős number ("None" if not computed)
31/// - `is_erdos`: "true" for Paul Erdős himself, "false" otherwise
32/// - `aliases`: JSON array of alternative name spellings/forms
33/// - `updated_at`: Unix timestamp of last update
34///
35/// # Returns
36/// A HashMap with default/placeholder values for all properties
37#[allow(unused)]
38pub fn person_properties() -> HashMap<String, String> {
39    let mut props = HashMap::new();
40    props.insert("name".to_string(), "".to_string());
41    props.insert("erdos_number".to_string(), "None".to_string());
42    props.insert("is_erdos".to_string(), "false".to_string());
43    props.insert("aliases".to_string(), "[]".to_string());
44    props.insert("updated_at".to_string(), "0".to_string());
45    props
46}
47
48/// Returns the property schema for Publication vertices.
49///
50/// # Properties
51/// - `title`: Full title of the publication
52/// - `year`: Publication year as a string
53/// - `venue`: Journal, conference, or other publication venue
54/// - `publication_id`: Source-specific identifier (e.g., ArXiv ID, DBLP key)
55///
56/// # Returns
57/// A HashMap with default/placeholder values for all properties
58#[allow(unused)]
59pub fn publication_properties() -> HashMap<String, String> {
60    let mut props = HashMap::new();
61    props.insert("title".to_string(), "".to_string());
62    props.insert("year".to_string(), "0".to_string());
63    props.insert("venue".to_string(), "".to_string());
64    props.insert("publication_id".to_string(), "".to_string());
65    props
66}
67
68/// Returns the property schema for COAUTHORED_WITH edges.
69///
70/// # Properties
71/// - `weight`: Number of papers co-authored (starts at 1, increments with each collaboration)
72/// - `publication_ids`: JSON array of publication IDs representing the collaborations
73///
74/// # Returns
75/// A HashMap with default/placeholder values for all properties
76#[allow(unused)]
77pub fn coauthored_with_properties() -> HashMap<String, String> {
78    let mut props = HashMap::new();
79    props.insert("weight".to_string(), "1".to_string());
80    props.insert("publication_ids".to_string(), "[]".to_string());
81    props
82}
