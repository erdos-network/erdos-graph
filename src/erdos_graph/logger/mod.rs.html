25/// Global logger facade.
26///
27/// Call `init_logger` early in `main` to set a concrete logger. Tests may use
28/// `set_logger_for_tests` to install a test logger.
29static mut GLOBAL_LOGGER: Option<&'static dyn Logger> = None;
30
31/// Initialize the global logger for the lifetime of the program.
32/// Returns a guard-like value in the future; for now this sets a static.
33pub fn init_logger<L: Logger>(logger: L) {
34    // Leak the logger so it can be referenced via a static pointer safely.
35    // This is a deliberate choice for a global singleton with program lifetime.
36    let boxed: Box<dyn Logger> = Box::new(logger);
37    let leaked: &'static dyn Logger = Box::leak(boxed);
38    unsafe {
39        GLOBAL_LOGGER = Some(leaked);
40    }
41}
42
43/// For tests: set a logger that will be used by the global facade.
44pub fn set_logger_for_tests<L: Logger>(logger: L) {
45    init_logger(logger);
46}
47
48/// Log using the global logger if set, otherwise no-op.
49pub fn log(level: LogLevel, message: &str) {
50    unsafe {
51        if let Some(logger) = GLOBAL_LOGGER {
52            logger.log(level, message);
53        }
54    }
55}
56
57/// Convenience functions
58pub fn info(msg: &str) {
59    log(LogLevel::Info, msg);
60}
61
62pub fn debug(msg: &str) {
63    log(LogLevel::Debug, msg);
64}
65
66pub fn error(msg: &str) {
67    log(LogLevel::Error, msg);
