1//! Custom logging primitives for the Erd≈ës Graph project.
2//!
3//! Responsibilities:
4//! - Provide a lightweight `Logger` trait used across crates
5//! - Offer a baseline no-op implementation for tests and benchmarking
6//! - Centralize log level semantics without pulling a full logging framework
7//!

13#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
14pub enum LogLevel {
15    Trace,
16    Debug,
17    Info,
18    Warn,
19    Error,
20}
21
22impl LogLevel {
23    /// Return a short string representation suitable for logs.
24    pub fn as_str(&self) -> &'static str {
25        match self {
26            LogLevel::Trace => "TRACE",
27            LogLevel::Debug => "DEBUG",
28            LogLevel::Info => "INFO",
29            LogLevel::Warn => "WARN",
30            LogLevel::Error => "ERROR",
31        }
32    }
33}

35/// Minimal logger interface used throughout the project.
36///
37/// Implementors should be cheap to clone and thread-safe if used in multithreaded contexts.
38/// For stubs, implement `log` and optionally override convenience methods.
39/// TODO: consider adding `Send + Sync` bounds once we finalize the threading model.
40/// The logger trait used across the project.
41///
42/// Implementations must be `Send + Sync` so they can be shared across threads
43/// when the global facade is used.
44pub trait Logger: Send + Sync + 'static {
45    /// Emit a log record at the given level.
46    fn log(&self, level: LogLevel, message: &str);
47
48    /// Flush any buffered records.
49    fn flush(&self) {}
50
51    /// Convenience: trace level.
52    fn trace(&self, message: &str) {
53        self.log(LogLevel::Trace, message);
54    }
55    /// Convenience: debug level.
56    fn debug(&self, message: &str) {
57        self.log(LogLevel::Debug, message);
58    }
59    /// Convenience: info level.
60    fn info(&self, message: &str) {
61        self.log(LogLevel::Info, message);
62    }
63    /// Convenience: warn level.
64    fn warn(&self, message: &str) {
65        self.log(LogLevel::Warn, message);
66    }
67    /// Convenience: error level.
68    fn error(&self, message: &str) {
69        self.log(LogLevel::Error, message);
70    }
71}
72
73/// No-op logger used by default in tests and when logging is disabled.
74#[derive(Debug, Default, Clone, Copy)]
75pub struct NoopLogger;
76
77impl Logger for NoopLogger {
78    fn log(&self, _level: LogLevel, _message: &str) {
79        // intentionally no-op
80    }
81}

83/// Very small stdout logger for quick debugging.
84///
85/// NOTE: This is a stub and formatting is intentionally minimal.
86/// TODO: Add timestamps and consistent formatting; gate by minimum level.
87#[derive(Debug, Default, Clone, Copy)]
88pub struct StdoutLogger;
89
90impl Logger for StdoutLogger {
91    fn log(&self, level: LogLevel, message: &str) {
92        // Emit a small JSON object to stdout so logs are easier to parse
93        // by structured log collectors. Keep the shape minimal for now.
94        let ts = chrono::Utc::now().to_rfc3339();
95        // Example: {"ts":"...","level":"INFO","msg":"..."}
96        let json = serde_json::json!({
97            "ts": ts,
98            "level": level.as_str(),
99            "msg": message,
100        });
101        println!("{}", json);
102    }
103
104    fn flush(&self) {
105        // stdout is line-buffered; nothing to do for the stub
106    }
107}
