46    /// Preloads all edges into the bloom filter
47    pub(crate) fn preload_edge_bloom(
48        &mut self,
49        engine: &Arc<HelixGraphEngine>,
50    ) -> Result<(), Box<dyn std::error::Error>> {
51        if self.edge_cache.edges_loaded {
52            return Ok(());
53        }
54        let start = Instant::now();
55        let mut count = 0;
56        let txn = engine.storage.graph_env.read_txn()?;
57        let arena = Bump::new();
58
59        if let Ok(iter) = engine.storage.edges_db.iter(&txn) {
60            for (edge_id, edge_bytes) in iter.flatten() {
61                if let Ok(edge) = Edge::from_bincode_bytes(edge_id, edge_bytes, &arena)
62                    && edge.label == crate::db::schema::COAUTHORED_WITH_TYPE
63                {
64                    let u = edge.from_node;
65                    let v = edge.to_node;
66                    self.edge_cache.cold_bloom.set(&(u, v));
67                    self.edge_cache.cold_bloom.set(&(v, u));
68                    count += 1;
69                }
70            }
71        }
72
73        self.edge_cache.edges_loaded = true;
74        logger::info(&format!(
75            "Preloaded {} edges into Bloom filter in {}ms",
76            count,
77            start.elapsed().as_millis()
78        ));
79        Ok(())
80    }

82    /// Preloads all author vertices into the cache
83    pub(crate) fn preload_authors(
84        &mut self,
85        engine: &Arc<HelixGraphEngine>,
86    ) -> Result<(), Box<dyn std::error::Error>> {
87        let start = Instant::now();
88        let mut count = 0;
89        let txn = engine.storage.graph_env.read_txn()?;
90
91        if let Some((name_index_db, _)) = engine.storage.secondary_indices.get("name")
92            && let Ok(iter) = name_index_db.iter(&txn)
93        {
94            for (key_bytes, node_id) in iter.flatten() {
95                if let Ok(HelixValue::String(name)) = bincode::deserialize(key_bytes) {
96                    let uuid = Uuid::from_u128(node_id);
97                    let norm_name = normalize_author(&name);
98                    let v = GraphVertex::with_id(uuid, crate::db::schema::PERSON_TYPE)
99                        .property("name", name);
100                    self.author_cache.insert(norm_name, v);
101                    count += 1;
102                }
103            }
104        }
105
106        logger::info(&format!(
107            "Preloaded {} authors in {}ms",
108            count,
109            start.elapsed().as_millis()
110        ));
111        Ok(())
112    }

114    /// Prefetch specific edge edges
115    pub(crate) fn prefetch_coauthor_edges(
116        &mut self,
117        edge_keys: &[(u128, u128)],
118        engine: &Arc<HelixGraphEngine>,
119    ) -> Result<(), Box<dyn std::error::Error>> {
120        if edge_keys.is_empty() {
121            return Ok(());
122        }
123
124        let txn = engine.storage.graph_env.read_txn()?;
125        let arena = Bump::new();
126
127        use crate::db::schema::COAUTHORED_WITH_TYPE;
128        let label_hash = helix_db::utils::label_hash::hash_label(COAUTHORED_WITH_TYPE, None);
129
130        // Optimization: Group by source to avoid creating an iterator for every pair
131        let mut adjacency_requests: HashMap<u128, HashSet<u128>> = HashMap::new();
132        for &(u, v) in edge_keys {
133            if self.edge_cache.get((u, v)).is_none() && self.edge_cache.exists((u, v)) {
134                adjacency_requests.entry(u).or_default().insert(v);
135            }
136        }
137
138        for (u, targets) in adjacency_requests {
139            let out_key = HelixGraphStorage::out_edge_key(&u, &label_hash);
140
141            if let Ok(iter) = engine.storage.out_edges_db.prefix_iter(&txn, &out_key) {
142                for (_, val_bytes) in iter.flatten() {
143                    if val_bytes.len() >= 32 {
144                        let mut to_node_bytes = [0u8; 16];
145                        to_node_bytes.copy_from_slice(&val_bytes[16..32]);
146                        let to_node_id = u128::from_be_bytes(to_node_bytes);
147
148                        if targets.contains(&to_node_id) {
149                            let mut edge_id_bytes = [0u8; 16];
150                            edge_id_bytes.copy_from_slice(&val_bytes[0..16]);
151                            let edge_id = u128::from_be_bytes(edge_id_bytes);
152
153                            if let Ok(Some(edge_bytes)) =
154                                engine.storage.edges_db.get(&txn, &edge_id)
155                                && let Ok(edge) =
156                                    Edge::from_bincode_bytes(edge_id, edge_bytes, &arena)
157                                && let Some(props) = edge.properties
158                                && let Some(w_val) = props.get("weight")
159                            {
160                                let weight = match w_val {
161                                    HelixValue::U64(w) => *w,
162                                    HelixValue::I64(w) => *w as u64,
163                                    HelixValue::F64(w) => *w as u64,
164                                    _ => 1,
165                                };
166                                self.edge_cache.put((u, to_node_id), weight);
167                            }
168                        }
169                    }
170                }
171            }
172        }
173
174        Ok(())
175    }

178fn json_to_helix_value(v: &Value) -> HelixValue {
179    match v {
180        Value::Null => HelixValue::Empty,
181        Value::Bool(b) => HelixValue::Boolean(*b),
182        Value::Number(n) => {
183            if let Some(i) = n.as_i64() {
184                HelixValue::I64(i)
185            } else if let Some(u) = n.as_u64() {
186                HelixValue::U64(u)
187            } else if let Some(f) = n.as_f64() {
188                HelixValue::F64(f)
189            } else {
190                HelixValue::F64(0.0)
191            }
192        }
193        Value::String(s) => HelixValue::String(s.clone()),
194        Value::Array(arr) => {
195            let vec: Vec<HelixValue> = arr.iter().map(json_to_helix_value).collect();
196            HelixValue::Array(vec)
197        }
198        Value::Object(obj) => {
199            let map: HashMap<String, HelixValue> = obj
200                .iter()
201                .map(|(k, v)| (k.clone(), json_to_helix_value(v)))
202                .collect();
203            HelixValue::Object(map)
204        }
205    }
206}

208fn value_to_string(v: &HelixValue) -> String {
209    match v {
210        HelixValue::String(s) => s.clone(),
211        HelixValue::I64(i) => i.to_string(),
212        HelixValue::U64(u) => u.to_string(),
213        HelixValue::F64(f) => f.to_string(),
214        HelixValue::Boolean(b) => b.to_string(),
215        _ => String::new(),
216    }
217}

219/// Helper to convert our generic GraphVertex to Helix Node
220fn to_helix_node<'arena>(v: &GraphVertex, arena: &'arena Bump) -> Node<'arena> {
221    let items = v
222        .props
223        .iter()
224        .map(|(k, val)| (arena.alloc_str(k) as &str, json_to_helix_value(val)));
225
226    let props = if v.props.is_empty() {
227        None
228    } else {
229        Some(ImmutablePropertiesMap::new(v.props.len(), items, arena))
230    };
231
232    Node {
233        id: v.id.as_u128(),
234        label: arena.alloc_str(&v.t),
235        version: 1,
236        properties: props,
237    }
238}

240/// Helper to convert our generic GraphEdge to Helix Edge
241fn to_helix_edge<'arena>(e: &GraphEdge, id: u128, arena: &'arena Bump) -> Edge<'arena> {
242    let items = e
243        .props
244        .iter()
245        .map(|(k, val)| (arena.alloc_str(k) as &str, json_to_helix_value(val)));
246
247    let props = if e.props.is_empty() {
248        None
249    } else {
250        Some(ImmutablePropertiesMap::new(e.props.len(), items, arena))
251    };
252
253    Edge {
254        id,
255        label: arena.alloc_str(&e.t),
256        version: 1,
257        from_node: e.source_id.as_u128(),
258        to_node: e.target_id.as_u128(),
259        properties: props,
260    }

263/// Flushes the write buffer and pending edge updates to the database.
264pub(crate) fn flush_buffer(
265    context: &mut IngestionContext,
266    engine: &Arc<HelixGraphEngine>,
267) -> Result<(), Box<dyn std::error::Error>> {
268    if context.write_buffer.is_empty() && context.pending_edge_updates.is_empty() {
269        return Ok(());
270    }
271
272    let start = Instant::now();
273    let mut count = context.write_buffer.len();
274    let mut wtxn = engine.storage.graph_env.write_txn()?;
275    let arena = Bump::new();
276
277    // 1. Convert pending edge updates to WriteOperations
278    // We need to fetch current weights from DB if they weren't in cache
279    for ((u_id, v_id), increment) in context.pending_edge_updates.drain() {
280        let key = (u_id, v_id);
281
282        // Determine final weight
283        let final_weight = if let Some(cached) = context.edge_cache.get(key) {
284            // If it was in cache, the cache already has the incremented value
285            // because we updated it in create_coauthor_edge
286            cached
287        } else {
288            // Not in hot cache, check DB
289            // Note: In a real implementation we might want to batch this read too,
290            // but for now we trust the "prefetch_coauthor_edges" did its job
291            // or we take the hit for cold edges.
292            // Since we updated cache.put(key, 1) or similar in create_coauthor_edge
293            // for new edges, this branch is mostly for "exists in bloom but not LRU" cases.
294            match fetch_weight_from_db(u_id, v_id, engine, &wtxn)? {
295                Some(w) => w + increment,
296                None => increment,
297            }
298        };
299
300        // Update cache with final confirmed weight
301        context.edge_cache.put(key, final_weight);
302
303        // Create the edge operations
304        let u_uuid = Uuid::from_u128(u_id);
305        let v_uuid = Uuid::from_u128(v_id);
306
307        use crate::db::schema::COAUTHORED_WITH_TYPE;
308        let edge1 =
309            GraphEdge::new(u_uuid, v_uuid, COAUTHORED_WITH_TYPE).property("weight", final_weight);
310        context.write_buffer.push(WriteOperation::CreateEdge(edge1));
311
312        let edge2 =
313            GraphEdge::new(v_uuid, u_uuid, COAUTHORED_WITH_TYPE).property("weight", final_weight);
314        context.write_buffer.push(WriteOperation::CreateEdge(edge2));
315
316        count += 2;
317    }
318
319    for op in context.write_buffer.drain(..) {
320        match op {
321            WriteOperation::CreateVertex(v) => {
322                let node = to_helix_node(&v, &arena);
323                if let Ok(bytes) = bincode::serialize(&node) {
324                    engine.storage.nodes_db.put_with_flags(
325                        &mut wtxn,
326                        PutFlags::empty(),
327                        &node.id,
328                        &bytes,
329                    )?;
330
331                    for (idx_name, (db, _)) in &engine.storage.secondary_indices {
332                        if let Some(val) = node.get_property(idx_name) {
333                            let val_str = value_to_string(val);
334                            if let Ok(key_bytes) = bincode::serialize(&HelixValue::String(val_str))
335                            {
336                                db.put_with_flags(
337                                    &mut wtxn,
338                                    PutFlags::empty(),
339                                    &key_bytes,
340                                    &node.id,
341                                )?;
342                            }
343                        }
344                    }
345                }
346            }
347            WriteOperation::CreateEdge(e) => {
348                let edge_id = uuid::Uuid::now_v7().as_u128();
349                let edge = to_helix_edge(&e, edge_id, &arena);
350
351                if let Ok(bytes) = bincode::serialize(&edge) {
352                    engine.storage.edges_db.put_with_flags(
353                        &mut wtxn,
354                        PutFlags::APPEND,
355                        &edge.id,
356                        &bytes,
357                    )?;
358
359                    let label_hash = helix_db::utils::label_hash::hash_label(edge.label, None);
360
361                    engine.storage.out_edges_db.put(
362                        &mut wtxn,
363                        &HelixGraphStorage::out_edge_key(&edge.from_node, &label_hash),
364                        &HelixGraphStorage::pack_edge_data(&edge.id, &edge.to_node),
365                    )?;
366
367                    engine.storage.in_edges_db.put(
368                        &mut wtxn,
369                        &HelixGraphStorage::in_edge_key(&edge.to_node, &label_hash),
370                        &HelixGraphStorage::pack_edge_data(&edge.id, &edge.from_node),
371                    )?;
372                }
373            }
374        }
375    }
376
377    wtxn.commit()?;
378
379    logger::debug(&format!(
380        "Flushed {} operations in {}ms",
381        count,
382        start.elapsed().as_millis()
383    ));
384
385    Ok(())
386}

388fn fetch_weight_from_db(
389    u: u128,
390    v: u128,
391    engine: &Arc<HelixGraphEngine>,
392    txn: &heed3::RwTxn,
393) -> Result<Option<u64>, Box<dyn std::error::Error>> {
394    use crate::db::schema::COAUTHORED_WITH_TYPE;
395    let label_hash = helix_db::utils::label_hash::hash_label(COAUTHORED_WITH_TYPE, None);
396    let out_key = HelixGraphStorage::out_edge_key(&u, &label_hash);
397    let arena = Bump::new();
398
399    if let Ok(iter) = engine.storage.out_edges_db.prefix_iter(txn, &out_key) {
400        for (_, val_bytes) in iter.flatten() {
401            if val_bytes.len() >= 32 {
402                let mut to_node_bytes = [0u8; 16];
403                to_node_bytes.copy_from_slice(&val_bytes[16..32]);
404                let to_node_id = u128::from_be_bytes(to_node_bytes);
405
406                if to_node_id == v {
407                    let mut edge_id_bytes = [0u8; 16];
408                    edge_id_bytes.copy_from_slice(&val_bytes[0..16]);
409                    let edge_id = u128::from_be_bytes(edge_id_bytes);
410
411                    if let Ok(Some(edge_bytes)) = engine.storage.edges_db.get(txn, &edge_id)
412                        && let Ok(edge) = Edge::from_bincode_bytes(edge_id, edge_bytes, &arena)
413                        && let Some(props) = edge.properties
414                        && let Some(w_val) = props.get("weight")
415                    {
416                        return Ok(match w_val {
417                            HelixValue::U64(w) => Some(*w),
418                            HelixValue::I64(w) => Some(*w as u64),
419                            _ => Some(1),
420                        });
421                    }
422                }
423            }
424        }
425    }
426    Ok(None)
427}

429/// Ingests a batch of publication records into the database.
430pub(crate) async fn ingest_batch(
431    records: Vec<PublicationRecord>,
432    engine: Arc<HelixGraphEngine>,
433    config: &Config,
434    context: &mut IngestionContext,
435) -> Result<(), Box<dyn std::error::Error>> {
436    let _start = Instant::now();
437    if records.is_empty() {
438        return Ok(());
439    }
440
441    let mut new_records = Vec::with_capacity(records.len());
442    let mut batch_seen_ids = HashSet::new();
443
444    for record in records {
445        if batch_seen_ids.contains(&record.id) {
446            continue;
447        }
448
449        if !context
450            .dedup_cache
451            .check_exists_and_cache(&record, &engine, config)
452        {
453            new_records.push(record.clone());
454            batch_seen_ids.insert(record.id.clone());
455        }
456    }
457
458    if new_records.is_empty() {
459        return Ok(());
460    }
461
462    let mut batch_authors = HashSet::new();
463    for record in &new_records {
464        for author in &record.authors {
465            batch_authors.insert(author.clone());
466        }
467    }
468
469    for author in &batch_authors {
470        get_or_create_author_vertex(author, &mut context.write_buffer, &mut context.author_cache)?;
471    }
472
473    let mut record_author_vertices: Vec<Vec<GraphVertex>> = Vec::with_capacity(new_records.len());
474    let mut all_involved_authors: HashSet<Uuid> = HashSet::new();
475
476    for record in &new_records {
477        let pub_vertex = add_publication(record, &mut context.write_buffer)?;
478        context.dedup_cache.add(
479            record.year,
480            pub_vertex.id.to_string(),
481            record.title.clone(),
482            &record.authors,
483            record.id.clone(),
484        );
485
486        let mut authors_for_rec = Vec::new();
487        for author_name in &record.authors {
488            let norm = normalize_author(author_name);
489            if let Some(v) = context.author_cache.get(&norm) {
490                authors_for_rec.push(v.clone());
491                all_involved_authors.insert(v.id);
492                create_authored_edge(v, &pub_vertex, &mut context.write_buffer)?;
493            }
494        }
495        record_author_vertices.push(authors_for_rec);
496    }
497
498    let mut edges_to_prefetch = HashSet::new();
499    for authors in &record_author_vertices {
500        for i in 0..authors.len() {
501            for j in (i + 1)..authors.len() {
502                let id1 = authors[i].id.as_u128();
503                let id2 = authors[j].id.as_u128();
504                edges_to_prefetch.insert((id1, id2));
505                edges_to_prefetch.insert((id2, id1));
506            }
507        }
508    }
509
510    let edges_vec: Vec<(u128, u128)> = edges_to_prefetch.into_iter().collect();
511
512    if !edges_vec.is_empty()
513        && let Err(e) = context.prefetch_coauthor_edges(&edges_vec, &engine)
514    {
515        logger::error(&format!("Failed to prefetch batch co-author edges: {}", e));
516    }
517
518    for authors in record_author_vertices {
519        for i in 0..authors.len() {
520            for j in (i + 1)..authors.len() {
521                let author1 = &authors[i];
522                let author2 = &authors[j];
523                create_coauthor_edge(
524                    author1,
525                    author2,
526                    &mut context.pending_edge_updates,
527                    &mut context.edge_cache,
528                )?;
529            }
530        }
531    }
532
533    flush_buffer(context, &engine)?;
534
535    Ok(())
536}

538/// Helper to get or create author vertex
539pub(crate) fn get_or_create_author_vertex(
540    name: &str,
541    write_buffer: &mut Vec<WriteOperation>,
542    author_cache: &mut HashMap<String, GraphVertex>,
543) -> Result<GraphVertex, Box<dyn std::error::Error>> {
544    let norm_name = normalize_author(name);
545
546    if let Some(v) = author_cache.get(&norm_name) {
547        return Ok(v.clone());
548    }
549
550    let person_type = crate::db::schema::PERSON_TYPE;
551    let v = GraphVertex::new(person_type).property("name", name.to_string());
552
553    write_buffer.push(WriteOperation::CreateVertex(v.clone()));
554
555    author_cache.insert(norm_name, v.clone());
556    Ok(v)
557}

559/// Helper to add publication vertex
560pub(crate) fn add_publication(
561    record: &PublicationRecord,
562    write_buffer: &mut Vec<WriteOperation>,
563) -> Result<GraphVertex, Box<dyn std::error::Error>> {
564    let type_id = crate::db::schema::PUBLICATION_TYPE;
565    let mut v = GraphVertex::new(type_id)
566        .property("title", record.title.clone())
567        .property("year", record.year.to_string())
568        .property("publication_id", record.id.clone())
569        .property("source", record.source.clone());
570
571    if let Some(venue) = &record.venue {
572        v = v.property("venue", venue.clone());
573    }
574
575    write_buffer.push(WriteOperation::CreateVertex(v.clone()));
576
577    Ok(v)
578}

592/// Create co-author edge with weight (buffered)
593pub(crate) fn create_coauthor_edge(
594    author1: &GraphVertex,
595    author2: &GraphVertex,
596    pending_updates: &mut HashMap<(u128, u128), u64>,
597    edge_cache: &mut EdgeCacheSystem,
598) -> Result<(), Box<dyn std::error::Error>> {
599    let (u, v) = if author1.id < author2.id {
600        (author1, author2)
601    } else {
602        (author2, author1)
603    };
604
605    let key = (u.id.as_u128(), v.id.as_u128());
606
607    // Update local batch counter
608    *pending_updates.entry(key).or_insert(0) += 1;
609
610    // We also speculatively update the cache so subsequent lookups in this batch
611    // (if we were still reading from it) would be correct-ish, though strict consistency
612    // is handled by the flush.
613    let cached_weight = edge_cache.get(key);
614    if let Some(current_weight) = cached_weight {
615        edge_cache.put(key, current_weight + 1);
616    } else if edge_cache.exists(key) {
617        // If it exists in bloom but not LRU, we don't know the exact weight yet.
618        // We will resolve this during flush.
619        // For now, just mark it as "seen" in cache if you want, or do nothing.
620    } else {
621        // New edge entirely
622        edge_cache.put(key, 1);
623    }
624
625    Ok(())
626}
