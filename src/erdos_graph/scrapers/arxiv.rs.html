11/// Configuration for the ArXiv scraper.
12///
13/// A small configuration bag used by the ArXiv scraping helpers.
14///
15/// Important semantics
16///
17/// - Paging is controlled by `page_size` (how many results requested per page).
18///   The scraper treats a page that returns fewer than `page_size` items as an
19///   indication of the end of results.
20///
21/// - Concurrency is governed by `channel_size` (an advisory number of
22///   concurrent workers / channel buffer size used when parallelizing page
23///   fetch/processing). The current implementation stores this value and uses
24///   it for sizing; callers may use it to control parallelism.
25///
26/// - `delay_ms` is a polite delay inserted between each paged request and is
27///   expressed in milliseconds (ms). Use `Duration::from_millis(cfg.delay_ms)`
28///   when converting to a `Duration`.
29///
30/// - `Default::default()` reads environment variables at process startup to
31///   provide reproducible, deploy-friendly defaults. Environment variable
32///   names are documented below.
33///
34/// Fields (with default values and units)
35///
36/// - `base_url: String` — Default: `"http://export.arxiv.org/api/query"`.
37///   The HTTP endpoint used to query the ArXiv Atom API.
38///
39/// - `page_size: usize` — Default: `100`.
40///   Number of results requested per page (affects network request size and the
41///   scraper's detection of the end of results).
42///
43/// - `channel_size: usize` — Default: `8`.
44///   Advisory concurrency / channel buffer size (unit: number of workers).
45///
46/// - `delay_ms: u64` — Default: `200` (milliseconds).
47///   Milliseconds to sleep between paged requests to avoid hammering the API.
48///
49/// Environment variables read by `Default` (when present)
50///
51/// - `ARXIV_API_BASE` — overrides `base_url`.
52/// - `ARXIV_PAGE_SIZE` — overrides `page_size`.
53/// - `ARXIV_CHANNEL_SIZE` — overrides `channel_size`.
54/// - `ARXIV_DELAY_MS` — overrides `delay_ms` (value parsed as integer ms).
55///
56/// Example
57///
58/// ```rust
59/// # use chrono::Utc;
60/// # use erdos_graph::scrapers::arxiv::ArxivConfig;
61/// // build a deterministic config for tests or for a mock server
62/// let mut cfg = ArxivConfig::default();
63/// cfg.page_size = 50; // smaller page for faster test cycles
64/// cfg.delay_ms = 0; // disable politeness delay in unit tests when hitting a mock
65/// // pass `cfg` into `scrape_range_with_config_async` for deterministic behavior
66/// ```
67///
68/// Notes and runtime/testing caveats
69///
70/// - Prefer calling `scrape_range_with_config_async(start, end, cfg)` in
71///   tests and pointing `cfg.base_url` to a mock HTTP server. This keeps
72///   tests deterministic and avoids relying on live network access.
73///
74/// - Because `Default` reads environment variables, tests that rely on
75///   `ArxivConfig::default()` may be influenced by the test environment; if
76///   reproducibility is required, construct an explicit `ArxivConfig`.
77///
78/// - The scraper implements a streaming, sliding-window parser that keeps
79///   memory usage proportional to the largest in-flight page/entry rather
80///   than the whole feed. Very large individual entries still require memory
81///   proportional to their size.
82///
83/// - `channel_size` is currently advisory; if you plan to add parallel page
84///   fetch/processing, use this field to size the worker pool and channel
85///   buffers.
86pub struct ArxivConfig {
87    pub base_url: String,
88    pub page_size: usize,
89    pub channel_size: usize,
90    /// Delay between pages in milliseconds
91    pub delay_ms: u64,
92}

94/// Default construction for `ArxivConfig`.
95///
96/// This implementation reads optional environment variables to override the
97/// compile-time defaults. The variables are:
98/// - `ARXIV_API_BASE` — base API URL (default: `http://export.arxiv.org/api/query`)
99/// - `ARXIV_PAGE_SIZE` — page size (default: `100`)
100/// - `ARXIV_CHANNEL_SIZE` — advisory channel/worker size (default: `8`)
101/// - `ARXIV_DELAY_MS` — polite delay between pages in milliseconds (default: `200`)
102impl Default for ArxivConfig {
103    fn default() -> Self {
104        fn env_usize(name: &str, default: usize) -> usize {
105            std::env::var(name)
106                .ok()
107                .and_then(|s| s.parse().ok())
108                .unwrap_or(default)
109        }
110
111        let base = std::env::var("ARXIV_API_BASE")
112            .unwrap_or_else(|_| "http://export.arxiv.org/api/query".to_string());
113        ArxivConfig {
114            base_url: base,
115            page_size: env_usize("ARXIV_PAGE_SIZE", 100),
116            channel_size: env_usize("ARXIV_CHANNEL_SIZE", 8),
117            delay_ms: env_usize("ARXIV_DELAY_MS", 200) as u64,
118        }
119    }
120}

122// Normalize text nodes: strip CDATA markers and unescape XML entities.
123fn normalize_text(txt: &str) -> String {
124    let s = txt.trim();
125    if s.starts_with("<![CDATA[") && s.ends_with("]]>") {
126        let inner = &s[9..s.len() - 3];
127        return inner.trim().to_string();
128    }
129    if let Ok(cow) = quick_unescape(s) {
130        return cow.into_owned().trim().to_string();
131    }
132    s.to_string()
133}

135// Extract the `term` attribute (or namespaced variant) from a start element's
136// attributes. Returns the value if present.
137fn extract_primary_category_from_attrs(e: &BytesStart) -> Option<String> {
138    for att in e.attributes().with_checks(false).flatten() {
139        let key = att.key.as_ref();
140        if key == b"term" || key.ends_with(b"term") {
141            return Some(String::from_utf8_lossy(att.value.as_ref()).to_string());
142        }
143    }
144    None
145}

147// Returns true if the optional publication date parses and falls in
148// [start_date, end_date).
149fn published_in_range(
150    pubdate_opt: &Option<String>,
151    start_date: DateTime<Utc>,
152    end_date: DateTime<Utc>,
153) -> bool {
154    pubdate_opt
155        .as_ref()
156        .and_then(|s| DateTime::parse_from_rfc3339(s).ok())
157        .map(|dt| {
158            let dt_utc = dt.with_timezone(&Utc);
159            dt_utc >= start_date && dt_utc < end_date
160        })
161        .unwrap_or(false)
162}
163/// Find a subslice `needle` in `hay` and return the byte index if present.
164///
165/// This is a tiny, allocation-free byte-search helper used by the
166/// streaming/sliding-window logic. It returns `None` when `needle` is empty
167/// or longer than `hay` to avoid spurious matches and to mirror the previous
168/// local implementation used throughout this module.
169fn find_subslice(hay: &[u8], needle: &[u8]) -> Option<usize> {
170    if needle.is_empty() || hay.len() < needle.len() {
171        return None;
172    }
173    hay.windows(needle.len()).position(|w| w == needle)
174}

176/// Drain and parse any complete `<entry>...</entry>` slices from `buf_acc`.
177///
178/// This helper repeatedly searches `buf_acc` for the next `<entry` start and
179/// a matching `</entry>` end. For each complete slice found it calls
180/// `parse_entry_str` and, when a `PublicationRecord` is returned, pushes it
181/// into `results`. Consumed bytes (from the beginning of `buf_acc` up to the
182/// end of the matched `</entry>`) are removed from `buf_acc` to keep the
183/// buffer small and to mirror streaming consumption semantics.
184///
185/// The function is intentionally conservative: it only extracts *complete*
186/// entries (both start and end found). Partial entries spanning future
187/// chunks are left in `buf_acc` for later completion. This behaviour is
188/// essential for correctness when streaming arbitrary chunk boundaries from
189/// the network.
190fn extract_complete_entries(
191    buf_acc: &mut Vec<u8>,
192    start_date: DateTime<Utc>,
193    end_date: DateTime<Utc>,
194    results: &mut Vec<PublicationRecord>,
195) {
196    loop {
197        let start_opt = find_subslice(buf_acc, b"<entry");
198        if start_opt.is_none() {
199            break;
200        }
201        let start = start_opt.unwrap();
202        if let Some(rel_end) = find_subslice(&buf_acc[start..], b"</entry>") {
203            let end = start + rel_end + b"</entry>".len();
204            let entry_slice = &buf_acc[start..end];
205            let entry_str = String::from_utf8_lossy(entry_slice);
206
207            if let Some(rec) = parse_entry_str(&entry_str, start_date, end_date) {
208                results.push(rec);
209            }
210
211            buf_acc.drain(0..end);
212            continue;
213        } else {
214            break;
215        }
216    }
217}

219// Parse a single `<entry>...</entry>` slice (slice-backed) and return a
220// PublicationRecord if the entry's published date falls in [start_date, end_date).
221fn parse_entry_str(
222    entry_str: &str,
223    start_date: DateTime<Utc>,
224    end_date: DateTime<Utc>,
225) -> Option<PublicationRecord> {
226    let mut reader = Reader::from_str(entry_str);
227    let mut tmp = Vec::new();
228    let mut inside_entry = false;
229    let mut cur_id: Option<String> = None;
230    let mut cur_title: Option<String> = None;
231    let mut cur_published: Option<String> = None;
232    let mut cur_journal_ref: Option<String> = None;
233    let mut cur_primary_cat: Option<String> = None;
234    let mut cur_authors: Vec<String> = Vec::new();
235
236    // Cheap pre-scan fallback: if the entry slice contains a primary_category
237    // attribute, grab its term value early so it's available when we build the
238    // PublicationRecord (handles odd namespace placement in some feeds).
239    if cur_primary_cat.is_none()
240        && let Some(idx) = entry_str.find("primary_category")
241        && let Some(term_pos) = entry_str[idx..].find("term=\"")
242    {
243        let start = idx + term_pos + "term=\"".len();
244        if let Some(endpos) = entry_str[start..].find('"') {
245            cur_primary_cat = Some(entry_str[start..start + endpos].to_string());
246        }
247    }
248
249    // Main streaming XML event loop: iterate over XML events produced by
250    // `quick-xml` and populate the temporary fields for the current `<entry>`.
251    // We stop when we reach the end of the entry or encounter EOF/errors.
252    loop {
253        tmp.clear();
254        match reader.read_event_into(&mut tmp) {
255            Ok(Event::Start(ref e)) => {
256                let name = e.local_name();
257                match name.as_ref() {
258                    // Start of an `<entry>` element: flip the flag so subsequent
259                    // child elements are captured into the current record fields.
260                    b"entry" => inside_entry = true,
261                    // Capture the identifier text inside `<id>`.
262                    b"id" if inside_entry => {
263                        if let Ok(txt) = reader.read_text(e.name()) {
264                            // Normalize CDATA and XML escapes before storing.
265                            cur_id = Some(normalize_text(&txt));
266                        }
267                    }
268                    // Capture the title for the entry.
269                    b"title" if inside_entry => {
270                        if let Ok(txt) = reader.read_text(e.name()) {
271                            cur_title = Some(normalize_text(&txt));
272                        }
273                    }
274                    // Capture the publication timestamp (RFC3339 expected).
275                    b"published" if inside_entry => {
276                        if let Ok(txt) = reader.read_text(e.name()) {
277                            cur_published = Some(normalize_text(&txt));
278                        }
279                    }
280                    // Capture a `journal_ref` if present; this takes precedence
281                    // over primary category for the `venue` field when present.
282                    b"journal_ref" if inside_entry => {
283                        if let Ok(txt) = reader.read_text(e.name()) {
284                            cur_journal_ref = Some(normalize_text(&txt));
285                        }
286                    }
287                    // Author name elements: append to the authors list.
288                    b"name" if inside_entry => {
289                        if let Ok(txt) = reader.read_text(e.name()) {
290                            cur_authors.push(normalize_text(&txt));
291                        }
292                    }
293                    // Primary category element (possibly namespaced). Extract
294                    // its `term` attribute and store it as a fallback venue.
295                    b"arxiv:primary_category" | b"primary_category" if inside_entry => {
296                        if let Some(val) = extract_primary_category_from_attrs(e) {
297                            cur_primary_cat = Some(val);
298                        }
299                    }
300                    _ => {
301                        // Handle namespaced primary_category variants like
302                        // `{http://arxiv.org/schemas/atom}primary_category`. The
303                        // element local-name may include a namespace, so we match
304                        // on the suffix and reuse the attribute extractor.
305                        if inside_entry
306                            && name.as_ref().ends_with(b"primary_category")
307                            && let Some(val) = extract_primary_category_from_attrs(e)
308                        {
309                            cur_primary_cat = Some(val);
310                        }
311                    }
312                }
313            }
314            Ok(Event::End(ref e)) => {
315                // On End events, if we've reached the end of an `<entry>`,
316                // decide whether the entry's published date falls inside the
317                // requested window and, if so, build and return a
318                // `PublicationRecord` populated from the fields we've
319                // accumulated. If not included, we break to indicate no
320                // matching record was produced from this slice.
321                if e.local_name().as_ref() == b"entry" {
322                    let include = published_in_range(&cur_published, start_date, end_date);
323                    if include {
324                        let year = cur_published
325                            .as_ref()
326                            .and_then(|s| DateTime::parse_from_rfc3339(s).ok())
327                            .map(|d| d.year() as u32)
328                            .unwrap_or(0);
329                        return Some(PublicationRecord {
330                            id: cur_id.take().unwrap_or_default(),
331                            title: cur_title.take().unwrap_or_default(),
332                            authors: cur_authors.clone(),
333                            year,
334                            // Prefer journal reference; fall back to primary
335                            // category when journal_ref is absent.
336                            venue: cur_journal_ref.take().or(cur_primary_cat.take()),
337                            source: String::from("arxiv"),
338                        });
339                    }
340                    // No matching record for this entry, stop parsing this
341                    // slice and return None to the caller.
342                    break;
343                }
344            }
345            Ok(Event::Eof) => break,
346            Err(_) => break,
347            _ => {}
348        }
349    }
350
351    None
352}

355// placement cases seen in tests).

357// return a record; it's a best-effort recovery.

360// Helper used in tests: given a sequence of byte chunks (as might be
361// received from the network), extract and parse complete <entry>...</entry>
362// slices and return PublicationRecords that fall inside the given date
363// range. This mirrors the sliding-window logic in
364// `scrape_range_with_config_async` and is intentionally `pub(crate)` so
365// tests can exercise chunked inputs without needing a live HTTP server.
366#[cfg(test)]
367pub(crate) fn parse_entries_from_chunks(
368    chunks: &[Vec<u8>],
369    start_date: DateTime<Utc>,
370    end_date: DateTime<Utc>,
371) -> Vec<PublicationRecord> {
372    // Local imports not needed here; parsing delegated to `parse_entry_str`
373
374    let mut results = Vec::new();
375    let mut buf_acc: Vec<u8> = Vec::new();
376
377    for chunk in chunks {
378        buf_acc.extend_from_slice(chunk);
379        // Extract any complete entries from the accumulated buffer.
380        extract_complete_entries(&mut buf_acc, start_date, end_date, &mut results);
381    }
382
383    results
384}

386/// Asynchronously scrape publication records from ArXiv for a given date range.
387///
388/// This is the high-level async entry point that uses `ArxivConfig::default()`
389/// to obtain configuration (which may read environment variables). It fetches
390/// pages from the ArXiv Atom API and parses `<entry>` elements in a
391/// streaming, sliding-window manner so memory usage stays proportional to the
392/// largest in-flight entry rather than the whole feed.
393///
394/// Parameters
395/// - `start_date`, `end_date`: the inclusive/exclusive time window to include
396///   publications (entries with `published` in [start_date, end_date) are
397///   returned).
398///
399/// Returns
400/// - `Ok(Vec<PublicationRecord>)` on success with the matched records.
401/// - `Err(...)` on network, parsing, or runtime errors.
402///
403/// Notes
404/// - For tests or deterministic runs prefer `scrape_range_with_config_async`
405///   and pass a custom `ArxivConfig` (e.g., with `base_url` pointing at a
406///   mock server).
407#[coverage(off)]
408pub async fn scrape_range_async(
409    start_date: DateTime<Utc>,
410    end_date: DateTime<Utc>,
411) -> Result<Vec<PublicationRecord>, Box<dyn std::error::Error>> {
412    let cfg = ArxivConfig::default();
413    scrape_range_with_config_async(start_date, end_date, cfg).await
414}

416/// Asynchronously scrape publication records using an explicit configuration.
417///
418/// Use this variant when you need deterministic behaviour (for example in
419/// tests) or to customize scraping parameters such as `base_url`, `page_size`,
420/// `channel_size`, or `delay_ms`.
421///
422/// Parameters
423/// - `start_date`, `end_date`: include publications with `published` in
424///   [start_date, end_date).
425/// - `cfg`: an `ArxivConfig` controlling endpoint, paging, and polite delays.
426///
427/// Returns
428/// - `Ok(Vec<PublicationRecord>)` on success.
429/// - `Err(...)` on failure (network, parsing, runtime errors).
430pub async fn scrape_range_with_config_async(
431    start_date: DateTime<Utc>,
432    end_date: DateTime<Utc>,
433    cfg: ArxivConfig,
434) -> Result<Vec<PublicationRecord>, Box<dyn std::error::Error>> {
435    let client = Client::new();
436    let mut start: usize = 0;
437    let page_size: usize = cfg.page_size;
438    let _channel_size: usize = cfg.channel_size;
439    let delay_ms: u64 = cfg.delay_ms;
440
441    let mut results: Vec<PublicationRecord> = Vec::new();
442
443    loop {
444        let url = format!(
445            "{}?search_query=all&start={}&max_results={}",
446            cfg.base_url, start, page_size
447        );
448
449        // Sliding-window streaming parser: accumulate chunks and extract <entry> slices
450        let mut resp = client.get(&url).send().await?;
451        // Remember how many results we had before this page so we can decide
452        // when to stop (arXiv returns fewer than page_size when exhausted)
453        let prev_results_len = results.len();
454        let mut buf_acc: Vec<u8> = Vec::new();
455
456        loop {
457            match resp.chunk().await? {
458                Some(chunk) => buf_acc.extend_from_slice(&chunk),
459                None => break,
460            }
461
462            // Extract complete entries
463            extract_complete_entries(&mut buf_acc, start_date, end_date, &mut results);
464        }
465
466        // Process any remaining complete entries after EOF
467        extract_complete_entries(&mut buf_acc, start_date, end_date, &mut results);
468
469        // Be polite to arXiv: small delay between paged requests
470        sleep(Duration::from_millis(delay_ms)).await;
471
472        // If fewer entries were added than page_size, we've reached the end
473        let added = results.len() - prev_results_len;
474        if added < page_size {
475            break;
476        }
477
478        start += page_size;
479    }
480
481    Ok(results)
482}

484/// Blocking wrapper around the async scraper.
485///
486/// This function constructs a small Tokio runtime and runs
487/// `scrape_range_async` to provide a synchronous API for callers that cannot
488/// be async. Prefer using the async variants where possible; this wrapper is
489/// intended for compatibility and tests that require a blocking interface.
490pub fn scrape_range(
491    start_date: DateTime<Utc>,
492    end_date: DateTime<Utc>,
493) -> Result<Vec<PublicationRecord>, Box<dyn std::error::Error>> {
494    // Create a small runtime to run the async function synchronously.
495    let rt = tokio::runtime::Runtime::new()?;
496    rt.block_on(scrape_range_async(start_date, end_date))
497}

514    // the consumed bytes are removed from the buffer.
515    #[test]
516    fn test_extract_complete_entries_single_and_chunked() {
517        let start = Utc.with_ymd_and_hms(2025, 9, 1, 0, 0, 0).unwrap();
518        let end = Utc.with_ymd_and_hms(2025, 12, 1, 0, 0, 0).unwrap();
519
520        // Build an entry split across two chunks (simulate network chunking)
521        let pre =
522            b"<feed><entry><id>s1</id><published>2025-10-01T00:00:00Z</published><title>S".to_vec();
523        let post = b"mall</title></entry></feed>".to_vec();

526        let mut results: Vec<PublicationRecord> = Vec::new();

537        buf_acc.extend_from_slice(&post);

543        // Buffer should no longer contain the consumed entry bytes (may contain trailing feed markers)
544        assert!(buf_acc.len() < (pre.len() + post.len()));
