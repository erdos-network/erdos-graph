54// Normalize text nodes: strip CDATA markers and unescape XML entities.
55fn normalize_text(txt: &str) -> String {
56    let s = txt.trim();
57    if s.starts_with("<![CDATA[") && s.ends_with("]]>") {
58        let inner = &s[9..s.len() - 3];
59        return inner.trim().to_string();
60    }
61    if let Ok(cow) = quick_unescape(s) {
62        return cow.into_owned().trim().to_string();
63    }
64    s.to_string()
65}

67// Extract the `term` attribute (or namespaced variant) from a start element's
68// attributes. Returns the value if present.
69fn extract_primary_category_from_attrs(e: &BytesStart) -> Option<String> {
70    for att in e.attributes().with_checks(false).flatten() {
71        let key = att.key.as_ref();
72        if key == b"term" || key.ends_with(b"term") {
73            return Some(String::from_utf8_lossy(att.value.as_ref()).to_string());
74        }
75    }
76    None
77}

79/// Find a subslice `needle` in `hay` and return the byte index if present.
80///
81/// This is a tiny, allocation-free byte-search helper used by the
82/// streaming/sliding-window logic. It returns `None` when `needle` is empty
83/// or longer than `hay` to avoid spurious matches and to mirror the previous
84/// local implementation used throughout this module.
85fn find_subslice(hay: &[u8], needle: &[u8]) -> Option<usize> {
86    if needle.is_empty() || hay.len() < needle.len() {
87        return None;
88    }
89    hay.windows(needle.len()).position(|w| w == needle)
90}

92/// Drain and parse any complete `<entry>...</entry>` slices from `buf_acc`.
93///
94/// This helper repeatedly searches `buf_acc` for the next `<entry` start and
95/// a matching `</entry>` end. For each complete slice found it calls
96/// `parse_entry_str` and, when a `PublicationRecord` is returned, pushes it
97/// into `results`. Consumed bytes (from the beginning of `buf_acc` up to the
98/// end of the matched `</entry>`) are removed from `buf_acc` to keep the
99/// buffer small and to mirror streaming consumption semantics.
100///
101/// The function is intentionally conservative: it only extracts *complete*
102/// entries (both start and end found). Partial entries spanning future
103/// chunks are left in `buf_acc` for later completion. This behaviour is
104/// essential for correctness when streaming arbitrary chunk boundaries from
105/// the network.
106fn extract_complete_entries(
107    buf_acc: &mut Vec<u8>,
108    start_date: DateTime<Utc>,
109    end_date: DateTime<Utc>,
110    producer: &QueueProducer<PublicationRecord>,
111    count: &mut usize,
112) {
113    loop {
114        let start_opt = find_subslice(buf_acc, b"<entry");
115        if start_opt.is_none() {
116            // logger::debug("No <entry> start found in buffer");
117            break;
118        }
119        let start = start_opt.unwrap();
120        if let Some(rel_end) = find_subslice(&buf_acc[start..], b"</entry>") {
121            let end = start + rel_end + b"</entry>".len();
122            let entry_slice = &buf_acc[start..end];
123            let entry_str = String::from_utf8_lossy(entry_slice);
124
125            if let Some(rec) = parse_entry_str(&entry_str, start_date, end_date) {
126                if let Err(e) = producer.submit(rec) {
127                    logger::error(&format!("Failed to submit record: {}", e));
128                }
129                *count += 1;
130            }
131
132            buf_acc.drain(0..end);
133            continue;
134        } else {
135            break;
136        }
137    }
138}

140// Parse a single `<entry>...</entry>` slice (slice-backed) and return a
141// PublicationRecord if the entry's published date falls in [start_date, end_date).
142fn parse_entry_str(
143    entry_str: &str,
144    start_date: DateTime<Utc>,
145    end_date: DateTime<Utc>,
146) -> Option<PublicationRecord> {
147    let mut reader = Reader::from_str(entry_str);
148    let mut tmp = Vec::new();
149    let mut inside_entry = false;
150    let mut cur_id: Option<String> = None;
151    let mut cur_title: Option<String> = None;
152    let mut cur_published: Option<String> = None;
153    let mut cur_journal_ref: Option<String> = None;
154    let mut cur_primary_cat: Option<String> = None;
155    let mut cur_authors: Vec<String> = Vec::new();
156
157    // Pre-scan for primary_category attribute to handle odd namespace placement
158    if cur_primary_cat.is_none()
159        && let Some(idx) = entry_str.find("primary_category")
160        && let Some(term_pos) = entry_str[idx..].find("term=\"")
161    {
162        let start = idx + term_pos + "term=\"".len();
163        if let Some(endpos) = entry_str[start..].find('"') {
164            cur_primary_cat = Some(entry_str[start..start + endpos].to_string());
165        }
166    }
167
168    // Main XML event loop to populate entry fields
169    loop {
170        tmp.clear();
171        match reader.read_event_into(&mut tmp) {
172            Ok(Event::Start(ref e)) => {
173                let name = e.local_name();
174                match name.as_ref() {
175                    b"entry" => inside_entry = true,
176                    // Capture the identifier text inside `<id>`.
177                    b"id" if inside_entry => {
178                        if let Ok(txt) = reader.read_text(e.name()) {
179                            // Normalize CDATA and XML escapes before storing.
180                            cur_id = Some(normalize_text(&txt));
181                        }
182                    }
183                    // Capture the title for the entry.
184                    b"title" if inside_entry => {
185                        if let Ok(txt) = reader.read_text(e.name()) {
186                            cur_title = Some(normalize_text(&txt));
187                        }
188                    }
189                    // Capture the publication timestamp (RFC3339 expected).
190                    b"published" if inside_entry => {
191                        if let Ok(txt) = reader.read_text(e.name()) {
192                            cur_published = Some(normalize_text(&txt));
193                        }
194                    }
195                    // Capture a `journal_ref` if present; this takes precedence
196                    // over primary category for the `venue` field when present.
197                    b"journal_ref" if inside_entry => {
198                        if let Ok(txt) = reader.read_text(e.name()) {
199                            cur_journal_ref = Some(normalize_text(&txt));
200                        }
201                    }
202                    b"name" if inside_entry => {
203                        if let Ok(txt) = reader.read_text(e.name()) {
204                            cur_authors.push(normalize_text(&txt));
205                        }
206                    }
207                    // Primary category element (possibly namespaced). Extract
208                    // its `term` attribute and store it as a fallback venue.
209                    b"arxiv:primary_category" | b"primary_category" if inside_entry => {
210                        if let Some(val) = extract_primary_category_from_attrs(e) {
211                            cur_primary_cat = Some(val);
212                        }
213                    }
214                    _ => {
215                        // Handle namespaced primary_category variants
216                        if inside_entry
217                            && name.as_ref().ends_with(b"primary_category")
218                            && let Some(val) = extract_primary_category_from_attrs(e)
219                        {
220                            cur_primary_cat = Some(val);
221                        }
222                    }
223                }
224            }
225            Ok(Event::End(ref e)) => {
226                // On End events, if we've reached the end of an `<entry>`,
227                // build and return a `PublicationRecord` populated from the
228                // fields we've accumulated.
229                if e.local_name().as_ref() == b"entry" {
230                    // Check if published date is valid and within range
231                    let published_str = cur_published.as_ref()?;
232                    let published_dt = DateTime::parse_from_rfc3339(published_str).ok()?;
233                    let published_utc = published_dt.with_timezone(&Utc);
234
235                    if published_utc < start_date || published_utc >= end_date {
236                        return None;
237                    }
238
239                    let year = published_utc.year() as u32;
240
241                    return Some(PublicationRecord {
242                        id: cur_id.take().unwrap_or_default(),
243                        title: cur_title.take().unwrap_or_default(),
244                        authors: cur_authors.clone(),
245                        year,
246                        // Prefer journal reference; fall back to primary
247                        // category when journal_ref is absent.
248                        venue: cur_journal_ref.take().or(cur_primary_cat.take()),
249                        source: String::from("arxiv"),
250                    });
251                }
252            }
253            Ok(Event::Eof) => break,
254            Err(_) => break,
255            _ => {}
256        }
257    }
258
259    None
260}

263// placement cases seen in tests).
264// Note: this fallback runs only when the full parse above failed to
265// return a record; it's a best-effort recovery.
266// (Function end)

268// Helper used in tests: given a sequence of byte chunks (as might be
269// received from the network), extract and parse complete <entry>...</entry>
270// slices and return PublicationRecords that fall inside the given date
271// range. This mirrors the sliding-window logic in
272// `scrape_range_with_config_async` and is intentionally `pub(crate)` so
273// tests can exercise chunked inputs without needing a live HTTP server.
274#[cfg(test)]
275pub(crate) fn parse_entries_from_chunks(
276    chunks: &[Vec<u8>],
277    start_date: DateTime<Utc>,
278    end_date: DateTime<Utc>,
279) -> Vec<PublicationRecord> {
280    // Local imports not needed here; parsing delegated to `parse_entry_str`
281
282    let mut results = Vec::new();
283    let mut buf_acc: Vec<u8> = Vec::new();
284
285    let queue = ThreadSafeQueue::new(QueueConfig::default());
286    let producer = queue.create_producer();
287
288    let mut count = 0;
289    for chunk in chunks {
290        buf_acc.extend_from_slice(chunk);
291        // Extract any complete entries from the accumulated buffer.
292        extract_complete_entries(&mut buf_acc, start_date, end_date, &producer, &mut count);
293    }
294
295    // Drain queue
296    drop(producer);
297    while let Some(r) = queue.dequeue() {
298        results.push(r);
299    }
300
301    results
302}

304/// Asynchronously scrape publication records from ArXiv for a given date range.
305///
306/// This is the high-level async entry point that uses `ArxivConfig::default()`
307/// to obtain configuration (which may read environment variables). It fetches
308/// pages from the ArXiv Atom API and parses `<entry>` elements in a
309/// streaming, sliding-window manner so memory usage stays proportional to the
310/// largest in-flight entry rather than the whole feed.
311///
312/// Parameters
313/// - `start_date`, `end_date`: the inclusive/exclusive time window to include
314///   publications (entries with `published` in [start_date, end_date) are
315///   returned).
316///
317/// Returns
318/// - `Ok(Vec<PublicationRecord>)` on success with the matched records.
319/// - `Err(...)` on network, parsing, or runtime errors.
320///
321/// - `Err(...)` on network, parsing, or runtime errors.
322///
323/// Notes
324/// - For tests or deterministic runs prefer `scrape_range_with_config_async`
325///   and pass a custom `ArxivSourceConfig` (e.g., with `base_url` pointing at a
326///   mock server).
327#[coverage(off)]
328pub async fn scrape_range_async(
329    start_date: DateTime<Utc>,
330    end_date: DateTime<Utc>,
331    producer: QueueProducer<PublicationRecord>,
332) -> Result<(), Box<dyn std::error::Error>> {
333    let cfg = ArxivSourceConfig::default();
334    scrape_range_with_config_async(start_date, end_date, cfg, producer).await
335}

337/// Asynchronously scrape publication records using an explicit configuration.
338///
339/// Use this variant when you need deterministic behaviour (for example in
340/// tests) or to customize scraping parameters such as `base_url`, `page_size`,
341/// `channel_size`, or `delay_ms`.
342///
343/// Parameters
344/// - `start_date`, `end_date`: include publications with `published` in
345///   [start_date, end_date).
346/// - `cfg`: an `ArxivSourceConfig` controlling endpoint, paging, and polite delays.
347///
348/// Returns
349/// - `Ok(Vec<PublicationRecord>)` on success.
350/// - `Err(...)` on failure (network, parsing, runtime errors).
351pub async fn scrape_range_with_config_async(
352    start_date: DateTime<Utc>,
353    end_date: DateTime<Utc>,
354    cfg: ArxivSourceConfig,
355    producer: QueueProducer<PublicationRecord>,
356) -> Result<(), Box<dyn std::error::Error>> {
357    let client = Client::new();
358    let mut start: usize = 0;
359    let page_size: usize = cfg.page_size;
360    let _channel_size: usize = cfg.channel_size;
361    let delay_ms: u64 = cfg.delay_ms;
362
363    if start_date > end_date {
364        return Ok(());
365    }
366
367    loop {
368        // Format dates as YYYYMMDDHHMM
369        let start_str = start_date.format("%Y%m%d%H%M").to_string();
370        let end_str = end_date.format("%Y%m%d%H%M").to_string();
371
372        let url = format!(
373            "{}?search_query=lastUpdatedDate:[{}+TO+{}]&start={}&max_results={}",
374            cfg.base_url, start_str, end_str, start, page_size
375        );
376
377        logger::debug(&format!("Fetching ArXiv page: {}", url));
378
379        let mut resp = client.get(&url).send().await?;
380        // when to stop (arXiv returns fewer than page_size when exhausted)
381        // Remember how many results we processed in this page
382        let mut page_record_count = 0;
383        let mut buf_acc: Vec<u8> = Vec::new();
384
385        loop {
386            match resp.chunk().await? {
387                Some(chunk) => buf_acc.extend_from_slice(&chunk),
388                None => break,
389            }
390
391            extract_complete_entries(
392                &mut buf_acc,
393                start_date,
394                end_date,
395                &producer,
396                &mut page_record_count,
397            );
398        }
399
400        // Process any remaining complete entries after EOF
401        // Process any remaining complete entries after EOF
402        extract_complete_entries(
403            &mut buf_acc,
404            start_date,
405            end_date,
406            &producer,
407            &mut page_record_count,
408        );
409
410        sleep(Duration::from_millis(delay_ms)).await;
411
412        // If fewer entries were added than page_size, we've reached the end
413        drop(resp); // Close response to free resources
414
415        // If fewer entries were added than page_size, we've reached the end
416        logger::debug(&format!(
417            "Parsed {} records from this page",
418            page_record_count
419        ));
420        if page_record_count < page_size {
421            break;
422        }
423
424        start += page_size;
425    }
426
427    Ok(())
428}

430/// Blocking wrapper around the async scraper.
431///
432/// This function constructs a small Tokio runtime and runs
433/// `scrape_range_async` to provide a synchronous API for callers that cannot
434/// be async. Prefer using the async variants where possible; this wrapper is
435/// intended for compatibility and tests that require a blocking interface.
436pub fn scrape_range(
437    start_date: DateTime<Utc>,
438    end_date: DateTime<Utc>,
439) -> Result<Vec<PublicationRecord>, Box<dyn std::error::Error>> {
440    // Create a small runtime to run the async function synchronously.
441    let rt = tokio::runtime::Runtime::new()?;
442    let queue = ThreadSafeQueue::new(QueueConfig::default());
443    let producer = queue.create_producer();
444
445    // We run the async function synchronously using the runtime.
446    // We use an async block to move the producer into the future.
447    rt.block_on(async move { scrape_range_async(start_date, end_date, producer).await })?;
448
449    let mut results = Vec::new();
450
451    while let Some(record) = queue.dequeue() {
452        results.push(record);
453    }
454
455    Ok(results)
456}
