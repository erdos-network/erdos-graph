1//! DBLP (Database systems and Logic Programming) XML Scraper
2//!
3//! This module implements a scraper for the DBLP computer science bibliography database.
4//! DBLP provides comprehensive bibliographic information about computer science publications
5//! including journals, conference proceedings, and other academic publications.
6//!
7//! The scraper downloads and parses the complete DBLP XML dump (several GB compressed)
8//! to extract publication metadata within specified date ranges.
9//!
10//! # Data Source
11//! - URL: https://dblp.org/xml/dblp.xml.gz
12//! - Format: Compressed XML dump of the entire DBLP database
13//! - Size: Multiple gigabytes (compressed), much larger uncompressed
14//! - Update frequency: Updated regularly by DBLP maintainers
15//!
16//! # XML Structure
17//! The DBLP XML contains various publication types:
18//! - `<article>` - Journal articles
19//! - `<inproceedings>` - Conference/workshop papers
20//! - `<book>`, `<incollection>`, `<proceedings>` - Other publication types
21//!
22//! Each entry contains metadata like authors, title, year, venue, etc.
23

86/// Scrapes DBLP publication data for a specified date range.
87///
88/// This function downloads the complete DBLP XML dump and extracts publication records
89/// that fall within the given date range. The filtering is done by publication year
90/// since DBLP doesn't provide more granular date filtering in their XML dump.
91///
92/// # Arguments
93/// * `start_date` - The start of the date range (inclusive). Only the year component is used.
94/// * `end_date` - The end of the date range (inclusive). Only the year component is used.
95///
96/// # Returns
97/// A vector of `PublicationRecord` structs containing the scraped publication data.
98///
99/// # Errors
100/// Returns an error if:
101/// - Network request to download the XML dump fails
102/// - XML decompression fails
103/// - XML parsing encounters unrecoverable errors
104/// - Memory allocation fails (the XML dump is very large)
105///
106/// # Performance Notes
107/// - Downloads the entire DBLP XML dump (~2-4 GB compressed, ~20+ GB uncompressed)
108/// - Processes the XML in a streaming fashion to manage memory usage
109/// - Filtering by year range happens during parsing to avoid loading irrelevant data
110/// - Can take several minutes to complete depending on network speed and system performance
111///
112/// # Example
113/// ```rust,no_run
114/// use erdos_graph::scrapers::dblp::scrape_range;
115/// use chrono::{TimeZone, Utc};
116///
117/// # tokio_test::block_on(async {
118/// let start = Utc.with_ymd_and_hms(2020, 1, 1, 0, 0, 0).unwrap();
119/// let end = Utc.with_ymd_and_hms(2021, 12, 31, 23, 59, 59).unwrap();
120/// let records = scrape_range(start, end).await?;
121/// # Ok::<(), Box<dyn std::error::Error>>(())
122/// # });
123/// ```
124pub async fn scrape_range(
125    start_date: DateTime<Utc>,
126    end_date: DateTime<Utc>,
127) -> Result<Vec<PublicationRecord>, Box<dyn std::error::Error>> {
128    scrape_range_with_config(start_date, end_date, DblpConfig::default()).await
129}

131/// Scrapes DBLP publication data with custom configuration (for testing).
132///
133/// This is an internal function that allows overriding the DBLP URL for testing purposes.
134/// Use `scrape_range` for production code.
135#[doc(hidden)]
136pub async fn scrape_range_with_config(
137    start_date: DateTime<Utc>,
138    end_date: DateTime<Utc>,
139    config: DblpConfig,
140) -> Result<Vec<PublicationRecord>, Box<dyn std::error::Error>> {
141    if start_date >= end_date {
142        return Ok(Vec::new());
143    }
144
145    // Extract year components for filtering - DBLP filtering is done by year only
146    let start_year = start_date.year();
147    let end_year = end_date.year();
148
149    // Initialize HTTP client for downloading the DBLP XML dump
150    let client = Client::new();
151    let url = &config.base_url;
152
153    // Download the complete DBLP XML dump
154    // WARNING: This file is very large (multiple GB), so this operation can take time
155    println!("Downloading DBLP XML dump from {}...", url);
156    let response = client.get(url).send().await?;
157
158    if !response.status().is_success() {
159        return Err(format!("Failed to download DBLP XML: HTTP {}", response.status()).into());
160    }
161
162    println!("Download complete, reading response body...");
163    let body = response.bytes().await?;
164
165    // Set up the decompression pipeline:
166    // 1. Wrap the downloaded bytes in a Cursor for reading
167    // 2. Create a GzDecoder to decompress the gzipped content
168    // 3. Wrap in BufReader for efficient buffered reading (required by quick_xml)
169    println!("Decompressing XML data...");
170    let cursor = Cursor::new(body.to_vec());
171    let decoder = GzDecoder::new(cursor);
172    let buf_reader = BufReader::new(decoder);
173
174    // Initialize the XML streaming parser
175    let mut reader = Reader::from_reader(buf_reader);
176    // Note: trim_text is not available in the current quick-xml version
177    // We handle whitespace trimming manually as needed
178
179    // Initialize collections for results and parsing
180    let mut records = Vec::new(); // Store successfully parsed publication records
181    let mut buf = Vec::new(); // Reusable buffer for XML event parsing
182
183    println!(
184        "Starting XML parsing for years {} to {}...",
185        start_year, end_year
186    );
187
188    // Main parsing loop - stream through the XML document
189    loop {
190        match reader.read_event_into(&mut buf)? {
191            Event::Start(ref e) => {
192                // Convert tag name from bytes to string for processing
193                let tag_name_bytes = e.name().as_ref().to_vec();
194                let tag_name = std::str::from_utf8(&tag_name_bytes)?;
195
196                // We're only interested in publication entries: articles and inproceedings
197                // - article: Journal papers, magazine articles
198                // - inproceedings: Conference papers, workshop papers
199                // Other types (book, incollection, proceedings, etc.) are skipped for now
200                if tag_name == "article" || tag_name == "inproceedings" {
201                    // Extract the DBLP key from the XML element's "key" attribute
202                    // The DBLP key is a unique identifier with format like:
203                    // - "journals/jacm/AuthorYear" for journal articles
204                    // - "conf/icml/AuthorYear" for conference papers
205                    let mut dblp_key = String::new();
206                    for attr in e.attributes() {
207                        let attr = attr?;
208                        if std::str::from_utf8(attr.key.as_ref())? == "key" {
209                            dblp_key = String::from_utf8(attr.value.to_vec())?;
210                            break;
211                        }
212                    }
213
214                    // Parse the XML element content to extract publication metadata
215                    match parse_dblp_element(&mut reader, tag_name, &dblp_key) {
216                        Ok(Some(record)) => {
217                            // Apply year-based filtering to only include records in our date range
218                            if record.year >= start_year as u32 && record.year <= end_year as u32 {
219                                records.push(record);
220
221                                // Progress indicator for large datasets
222                                if records.len() % 1000 == 0 {
223                                    println!(
224                                        "Processed {} matching records so far...",
225                                        records.len()
226                                    );
227                                }
228                            }
229                        }
230                        Ok(None) => {
231                            // Record was parsed but incomplete (missing required fields)
232                            // This is normal and expected for some DBLP entries
233                        }
234                        Err(e) => {
235                            // Non-fatal parsing error - log and continue with other records
236                            eprintln!(
237                                "Warning: Error parsing DBLP element with key '{}': {}",
238                                dblp_key, e
239                            );
240                            // Continue processing - individual record failures shouldn't stop the entire scrape
241                        }
242                    }
243                }
244                // All other XML elements are ignored (books, proceedings, etc.)
245            }
246            Event::Eof => {
247                // Reached end of XML document
248                println!("Finished parsing XML document.");
249                break;
250            }
251            _ => {
252                // Ignore other XML events (Text, End, Comment, etc.) at the top level
253                // We only care about Start events for publication elements
254            }
255        }
256
257        // Clear the buffer for reuse in the next iteration
258        // This is important for memory management when processing large XML files
259        buf.clear();
260    }
261
262    println!(
263        "DBLP scraping complete. Found {} records in date range {} to {}.",
264        records.len(),
265        start_year,
266        end_year
267    );
268    Ok(records)
269}

271/// Parses a single DBLP XML element (article or inproceedings) into a PublicationRecord.
272///
273/// This function reads through a complete XML element, extracting metadata fields
274/// like authors, title, year, and venue information. It uses a depth-tracking
275/// approach to handle nested XML structures correctly.
276///
277/// # Arguments
278/// * `reader` - Mutable reference to the XML reader positioned after the opening tag
279/// * `element_type` - The type of element being parsed ("article" or "inproceedings")  
280/// * `key` - The DBLP key extracted from the element's key attribute
281///
282/// # Returns
283/// * `Ok(Some(record))` - Successfully parsed a complete publication record
284/// * `Ok(None)` - Element was parsed but missing required fields (title, authors, or year)
285/// * `Err(error)` - Fatal parsing error (malformed XML, encoding issues, etc.)
286///
287/// # XML Structure Handled
288/// ```xml
289/// <article key="journals/jacm/Smith21">
290///   <author>John Smith</author>
291///   <author>Jane Doe</author>
292///   <title>Example Paper Title</title>
293///   <year>2021</year>
294///   <journal>Journal of the ACM</journal>  <!-- for articles -->
295///   <booktitle>ICML</booktitle>            <!-- for inproceedings -->
296///   <!-- other optional fields like pages, volume, etc. -->
297/// </article>
298/// ```
299fn parse_dblp_element<R: BufRead>(
300    reader: &mut Reader<R>,
301    element_type: &str,
302    key: &str,
303) -> Result<Option<PublicationRecord>, Box<dyn std::error::Error>> {
304    // Initialize parsing state
305    let mut buf = Vec::new(); // Buffer for XML event parsing
306    let mut title = String::new(); // Publication title
307    let mut authors = Vec::new(); // List of author names
308    let mut year: Option<u32> = None; // Publication year
309    let mut venue: Option<String> = None; // Journal name or conference name
310    let mut depth = 1; // Track XML nesting depth for proper parsing
311
312    // Parse nested XML elements until we reach the closing tag for this publication
313    // We start at depth 1 since we're already inside the opening tag
314    loop {
315        match reader.read_event_into(&mut buf)? {
316            Event::Start(e) => {
317                // Encountered a nested XML element
318                depth += 1;
319                let tag_name_bytes = e.name().as_ref().to_vec();
320                let tag_name = std::str::from_utf8(&tag_name_bytes)?;
321
322                // Process known metadata fields
323                match tag_name {
324                    "author" => {
325                        // Extract author name - DBLP can have multiple <author> tags
326                        if let Ok(author_name) = read_text_content(reader, "author") {
327                            authors.push(author_name)
328                        }
329                        depth -= 1; // read_text_content consumed the end tag
330                    }
331                    "title" => {
332                        // Extract publication title - should be unique per entry
333                        if let Ok(title_text) = read_text_content(reader, "title") {
334                            title = title_text
335                        }
336                        depth -= 1;
337                    }
338                    "year" => {
339                        // Extract publication year and convert to integer
340                        if let Ok(year_text) = read_text_content(reader, "year") {
341                            year = year_text.parse().ok();
342                        }
343                        depth -= 1;
344                    }
345                    "journal" if element_type == "article" => {
346                        // For journal articles, extract journal name as venue
347                        if let Ok(journal_name) = read_text_content(reader, "journal") {
348                            venue = Some(journal_name);
349                        }
350                        depth -= 1;
351                    }
352                    "booktitle" if element_type == "inproceedings" => {
353                        // For conference papers, extract conference name as venue
354                        if let Ok(booktitle_name) = read_text_content(reader, "booktitle") {
355                            venue = Some(booktitle_name);
356                        }
357                        depth -= 1;
358                    }
359                    _ => {
360                        // Skip other XML elements we don't need (pages, volume, number, ee, etc.)
361                        // This efficiently moves past unneeded content without processing it
362                        skip_element(reader, tag_name)?;
363                        depth -= 1;
364                    }
365                }
366            }
367            Event::End(e) => {
368                // Encountered a closing XML tag
369                depth -= 1;
370                let tag_name_bytes = e.name().as_ref().to_vec();
371                let tag_name = std::str::from_utf8(&tag_name_bytes)?;
372
373                // Check if this is the closing tag for our main element
374                if tag_name == element_type && depth == 0 {
375                    // We've reached the end of this publication entry
376                    break;
377                }
378            }
379            Event::Eof => {
380                // Unexpected end of file while parsing - this shouldn't happen in well-formed XML
381                return Err("Unexpected EOF while parsing DBLP element".into());
382            }
383            _ => {
384                // Ignore other XML events (Text, Comment, etc.) at this level
385            }
386        }
387        buf.clear(); // Clear buffer for next iteration
388    }
389
390    // Validate that we have all required fields for a complete publication record
391    // DBLP entries sometimes lack essential information, so we filter them out
392    if title.trim().is_empty() || authors.is_empty() || year.is_none() {
393        return Ok(None); // Return None for incomplete records (not an error)
394    }
395
396    // Create and return a complete PublicationRecord
397    Ok(Some(PublicationRecord {
398        id: key.to_string(),             // Use DBLP key as unique identifier
399        title: title.trim().to_string(), // Clean up whitespace from title
400        authors: authors
401            .into_iter()
402            .map(|a: String| a.trim().to_string()) // Clean up whitespace from author names
403            .collect(),
404        year: year.unwrap(),        // Safe to unwrap due to validation above
405        venue,                      // Journal or conference name (optional)
406        source: "dblp".to_string(), // Mark this record as coming from DBLP
407    }))
408}

410/// Reads the text content from an XML element.
411///
412/// This function advances the XML reader through an element, collecting all text
413/// content until it reaches the element's closing tag. It handles both regular
414/// text nodes and CDATA sections, and properly unescapes XML entities.
415///
416/// # Arguments
417/// * `reader` - Mutable reference to the XML reader positioned after an opening tag
418/// * `element_name` - Name of the element being read (for error reporting and validation)
419///
420/// # Returns
421/// The concatenated text content of the element, with XML entities unescaped.
422///
423/// # Errors
424/// Returns an error if:
425/// - XML parsing fails
426/// - Encoding conversion fails  
427/// - Unexpected EOF is encountered
428/// - XML entity unescaping fails
429///
430/// # Example XML Handling
431/// ```xml
432/// <title>Graph Theory &amp; Algorithms</title>  → "Graph Theory & Algorithms"
433/// <author><![CDATA[Smith, J. & Doe, A.]]></author>  → "Smith, J. & Doe, A."
434/// ```
435fn read_text_content<R: BufRead>(
436    reader: &mut Reader<R>,
437    element_name: &str,
438) -> Result<String, Box<dyn std::error::Error>> {
439    let mut buf = Vec::new(); // Buffer for XML event parsing
440    let mut content = String::new(); // Accumulated text content
441
442    loop {
443        match reader.read_event_into(&mut buf)? {
444            Event::Text(e) => {
445                // Regular text content with XML entities that need unescaping
446                // e.g., "&amp;" -> "&", "&lt;" -> "<", "&gt;" -> ">"
447                // Convert bytes to string and handle XML entities
448                let text = std::str::from_utf8(e.as_ref())?;
449                content.push_str(text);
450            }
451            Event::CData(e) => {
452                // CDATA sections contain literal text that doesn't need unescaping
453                // This is used for text that might contain XML-like characters
454                content.push_str(std::str::from_utf8(e.as_ref())?);
455            }
456            Event::End(e) => {
457                // Check if this is the closing tag for our element
458                let tag_name_bytes = e.name().as_ref().to_vec();
459                let tag_name = std::str::from_utf8(&tag_name_bytes)?;
460                if tag_name == element_name {
461                    // Found the matching closing tag - we're done reading content
462                    break;
463                }
464                // If it's a different closing tag, ignore it (nested element)
465            }
466            Event::Eof => {
467                // Unexpected end of file - malformed XML
468                return Err(
469                    format!("Unexpected EOF while reading element '{}'", element_name).into(),
470                );
471            }
472            _ => {
473                // Ignore other XML events (Start tags of nested elements, Comments, etc.)
474                // We only care about the text content at this level
475            }
476        }
477        buf.clear(); // Clear buffer for next iteration
478    }
479
480    Ok(content)
481}

483/// Efficiently skips over an XML element and all its nested content.
484///
485/// This function is used to bypass XML elements we don't need to process,
486/// such as `<pages>`, `<volume>`, `<ee>` (electronic edition), etc.
487/// It uses depth tracking to handle arbitrarily nested XML structures.
488///
489/// # Arguments
490/// * `reader` - Mutable reference to the XML reader positioned after an opening tag
491/// * `element_name` - Name of the element being skipped (used for error reporting)
492///
493/// # Returns
494/// `Ok(())` when the element has been successfully skipped, or an error if
495/// XML parsing fails or unexpected EOF is encountered.
496///
497/// # Performance Notes
498/// This is much more efficient than parsing and then discarding content,
499/// especially for elements that might contain large amounts of nested data.
500///
501/// # Example Usage
502/// When we encounter `<pages>123-456</pages>` but don't need page information,
503/// this function will advance the reader past the entire element without
504/// processing its content.
505fn skip_element<R: BufRead>(
506    reader: &mut Reader<R>,
507    element_name: &str,
508) -> Result<(), Box<dyn std::error::Error>> {
509    let mut buf = Vec::new(); // Buffer for XML event parsing
510    let mut depth = 1; // Track nesting depth (start at 1 since we're inside the opening tag)
511
512    loop {
513        match reader.read_event_into(&mut buf)? {
514            Event::Start(_) => {
515                // Encountered a nested opening tag - increase depth
516                depth += 1;
517            }
518            Event::End(e) => {
519                // Encountered a closing tag - decrease depth
520                depth -= 1;
521
522                // Check if we've returned to our original depth (found matching closing tag)
523                if depth == 0 {
524                    let tag_name_bytes = e.name().as_ref().to_vec();
525                    let tag_name = std::str::from_utf8(&tag_name_bytes)?;
526                    if tag_name == element_name {
527                        // Successfully skipped the entire element
528                        break;
529                    }
530                }
531            }
532            Event::Eof => {
533                // Unexpected end of file while skipping - indicates malformed XML
534                return Err(
535                    format!("Unexpected EOF while skipping element '{}'", element_name).into(),
536                );
537            }
538            _ => {
539                // Ignore all other XML events (Text, Comment, etc.) - we're just skipping
540            }
541        }
542        buf.clear(); // Clear buffer for next iteration
543    }
544
545    Ok(())
