31/// Helper struct to manage deduplication caching.
32pub(crate) struct DeduplicationCache {
33    /// Track which years have been loaded into the Bloom filters
34    loaded_years: HashSet<u32>,
35    /// Global Bloom Filter for Titles
36    title_filter: Bloom<String>,
37    /// Global Bloom Filter for Authors
38    author_filter: Bloom<String>,
39    /// Cache of known publication IDs (external IDs like arxiv:...) to avoid DB lookups
40    published_ids: HashSet<String>,
41    /// Bloom filter for edges to reduce DB lookups
42    edge_filter: Bloom<String>,
43    /// Track if edges have been loaded
44    edges_loaded: bool,
45}
46
47impl DeduplicationCache {
48    pub(crate) fn new(bloom_size: usize) -> Self {
49        Self {
50            loaded_years: HashSet::new(),
51            title_filter: Bloom::new_for_fp_rate(bloom_size, 0.01).unwrap(),
52            author_filter: Bloom::new_for_fp_rate(bloom_size, 0.01).unwrap(),
53            published_ids: HashSet::new(),
54            edge_filter: Bloom::new_for_fp_rate(bloom_size * 10, 0.01).unwrap(),
55            edges_loaded: false,
56        }
57    }
58
59    /// Adds a record to the cache.
60    fn add(&mut self, _year: u32, _id: Uuid, title: String, authors: &[String], pub_id: String) {
61        self.title_filter.set(&normalize_title(&title));
62        self.published_ids.insert(pub_id);
63
64        for author in authors {
65            self.author_filter.set(&normalize_title(author));
66        }
67    }

69    /// Checks if a record exists in the cache (and populates cache from DB if needed).
70    fn check_exists_and_cache(
71        &mut self,
72        record: &PublicationRecord,
73        datastore: &Database<impl Datastore>,
74        config: &Config,
75    ) -> bool {
76        let start = Instant::now();
77        // Ensure cache is populated for this year
78        if !self.loaded_years.contains(&record.year) {
79            self.populate_year(record.year, datastore);
80        }
81
82        // Check exact ID match in memory first - FAST path
83        if self.published_ids.contains(&record.id) {
84            logger::debug(&format!(
85                "Cache hit (known ID) for '{}' - took {}ms",
86                record.title,
87                start.elapsed().as_millis()
88            ));
89            return true;
90        }
91
92        let record_title_norm = normalize_title(&record.title);
93
94        // Global Title Bloom filter check
95        if !self.title_filter.check(&record_title_norm) {
96            logger::debug(&format!(
97                "Cache miss (title bloom) for '{}' - took {}ms",
98                record.title,
99                start.elapsed().as_millis()
100            ));
101            return false;
102        }
103
104        // Global Author Bloom filter check (if record has authors)
105        if !record.authors.is_empty() {
106            let mut any_author_match = false;
107            for author in &record.authors {
108                if self.author_filter.check(&normalize_title(author)) {
109                    any_author_match = true;
110                    break;
111                }
112            }
113            if !any_author_match {
114                logger::debug(&format!(
115                    "Cache miss (author bloom) for '{}' - took {}ms",
116                    record.title,
117                    start.elapsed().as_millis()
118                ));
119                return false;
120            }
121        }
122
123        if let Ok(type_prop) = Identifier::new(crate::db::schema::PUBLICATION_TYPE)
124            && let Ok(year_prop) = Identifier::new("year")
125            && let Ok(title_prop) = Identifier::new("title")
126        {
127            let year_val = Json::new(json!(record.year.to_string()));
128            let q = RangeVertexQuery::new()
129                .t(type_prop)
130                .with_property_equal_to(year_prop, year_val);
131
132            if let Ok(q) = q
133                && let Ok(q_props) = q.properties()
134                && let Ok(results) = datastore.get(q_props)
135                && let Some(QueryOutputValue::VertexProperties(vps)) = results.first()
136            {
137                let cosine = Cosine::default();
138                let record_title_lower = record.title.to_lowercase();
139
140                for vp in vps {
141                    if let Some(val) = vp.props.iter().find(|p| p.name == title_prop)
142                        && let Some(title) = val.value.as_str()
143                    {
144                        // If normalized titles are identical, skip cosine
145                        if record_title_norm == normalize_title(title) {
146                            if check_authors_similarity(
147                                &vp.vertex.id,
148                                &record.authors,
149                                datastore,
150                                config.deduplication.author_similarity_threshold,
151                            ) {
152                                logger::debug(&format!(
153                                    "Cache hit (exact title) for '{}' - took {}ms",
154                                    record.title,
155                                    start.elapsed().as_millis()
156                                ));
157                                return true;
158                            }
159                            continue;
160                        }
161
162                        let db_title_lower = title.to_lowercase();
163                        let similarity =
164                            cosine.for_str(&record_title_lower, &db_title_lower).nval();
165
166                        if similarity >= config.deduplication.title_similarity_threshold
167                            && check_authors_similarity(
168                                &vp.vertex.id,
169                                &record.authors,
170                                datastore,
171                                config.deduplication.author_similarity_threshold,
172                            )
173                        {
174                            logger::debug(&format!(
175                                "Cache hit (similarity) for '{}' - took {}ms",
176                                record.title,
177                                start.elapsed().as_millis()
178                            ));
179                            return true;
180                        }
181                    }
182                }
183            }
184        }
185
186        logger::debug(&format!(
187            "Cache checked (no match) for '{}' - took {}ms",
188            record.title,
189            start.elapsed().as_millis()
190        ));
191        false
192    }

194    fn populate_year(&mut self, year: u32, datastore: &Database<impl Datastore>) {
195        let start = Instant::now();
196        // Fetch all papers for the year
197        if let Ok(type_prop) = Identifier::new(crate::db::schema::PUBLICATION_TYPE)
198            && let Ok(year_prop) = Identifier::new("year")
199            && let Ok(title_prop) = Identifier::new("title")
200            && let Ok(id_prop) = Identifier::new("publication_id")
201        {
202            let year_val = Json::new(json!(year.to_string()));
203
204            // We need vertex ID, title, and publiction ID
205            let q = RangeVertexQuery::new()
206                .t(type_prop)
207                .with_property_equal_to(year_prop, year_val);
208
209            if let Ok(q) = q
210                && let Ok(q_props) = q.properties()
211                && let Ok(results) = datastore.get(q_props)
212                && let Some(QueryOutputValue::VertexProperties(vps)) = results.first()
213            {
214                let mut count = 0;
215                let mut pub_ids = Vec::with_capacity(vps.len());
216
217                for vp in vps {
218                    let mut title_opt = None;
219                    let mut id_opt = None;
220
221                    for p in &vp.props {
222                        if p.name == title_prop {
223                            title_opt = p.value.as_str();
224                        } else if p.name == id_prop {
225                            id_opt = p.value.as_str();
226                        }
227                    }
228
229                    if let Some(title) = title_opt {
230                        let norm = normalize_title(title);
231                        self.title_filter.set(&norm);
232                        pub_ids.push(vp.vertex.id);
233                        count += 1;
234                    }
235
236                    if let Some(pid) = id_opt {
237                        self.published_ids.insert(pid.to_string());
238                    }
239                }
240
241                // Batch fetch authors
242                if !pub_ids.is_empty() {
243                    for chunk in pub_ids.chunks(1000) {
244                        let q = SpecificVertexQuery::new(chunk.to_vec());
245                        if let Ok(q_in) = q.inbound()
246                            && let Ok(q_out) = q_in.outbound()
247                            && let Ok(q_props) = q_out.properties()
248                            && let Ok(results) = datastore.get(q_props)
249                            && let Some(QueryOutputValue::VertexProperties(author_vps)) =
250                                results.first()
251                            && let Ok(name_prop) = Identifier::new("name")
252                        {
253                            for avp in author_vps {
254                                if let Some(val) = avp.props.iter().find(|p| p.name == name_prop)
255                                    && let Some(name) = val.value.as_str()
256                                {
257                                    self.author_filter.set(&normalize_title(name));
258                                }
259                            }
260                        }
261                    }
262                }
263
264                logger::info(&format!(
265                    "Populated filters for year {}: {} records in {}ms",
266                    year,
267                    count,
268                    start.elapsed().as_millis()
269                ));
270            }
271        }
272        self.loaded_years.insert(year);
273    }
274}

276/// Normalizes a title for Bloom filter and comparison.
277/// - Lowercases
278/// - Tokenizes by whitespace
279/// - Removes stop words
280/// - Removes non-alphanumeric characters
281/// - Concatenates
282pub(crate) fn normalize_title(title: &str) -> String {
283    let stop_words: HashSet<&str> = [
284        "a", "an", "and", "are", "as", "at", "be", "by", "for", "from", "has", "he", "in", "is",
285        "it", "its", "of", "on", "that", "the", "to", "was", "were", "will", "with",
286    ]
287    .iter()
288    .cloned()
289    .collect();
290
291    title
292        .to_lowercase()
293        .split_whitespace()
294        .filter(|token| !stop_words.contains(token))
295        .map(|token| {
296            token
297                .chars()
298                .filter(|c| c.is_alphanumeric())
299                .collect::<String>()
300        })
301        .collect::<String>()
302}

304/// Normalizes an author name for cache and deduplication.
305/// - Lowercases
306/// - Splits by whitespace and rejoins with single space
307pub(crate) fn normalize_author(name: &str) -> String {
308    name.trim()
309        .to_lowercase()
310        .split_whitespace()
311        .collect::<Vec<_>>()
312        .join(" ")
313}

315/// Context for ingestion to maintain caches
316pub(crate) struct IngestionContext {
317    pub dedup_cache: DeduplicationCache,
318    pub author_cache: HashMap<String, Vertex>,
319    pub edge_cache: HashMap<(Uuid, Uuid), u64>,
320    pub write_buffer: Vec<WriteOperation>,
321}

323impl IngestionContext {
324    pub(crate) fn new(bloom_size: usize) -> Self {
325        Self {
326            dedup_cache: DeduplicationCache::new(bloom_size),
327            author_cache: HashMap::new(),
328            edge_cache: HashMap::new(),
329            write_buffer: Vec::with_capacity(1000),
330        }
331    }

333    /// Preloads all edges into the bloom filter
334    fn preload_edge_bloom(
335        &mut self,
336        datastore: &Database<impl Datastore>,
337    ) -> Result<(), Box<dyn std::error::Error>> {
338        if self.dedup_cache.edges_loaded {
339            return Ok(());
340        }
341
342        let start = Instant::now();
343        logger::info("Preloading edge bloom filter...");
344
345        if let Ok(edge_type) = Identifier::new(crate::db::schema::COAUTHORED_WITH_TYPE) {
346            let q = RangeVertexQuery::new();
347            let q_out = q.outbound()?;
348            let q_edges = q_out.t(edge_type);
349            if let Ok(results) = datastore.get(q_edges)
350                && let Some(QueryOutputValue::Edges(edges)) = results.first()
351            {
352                let count = edges.len();
353                for edge in edges {
354                    let edge_key = format!("{}-{}", edge.outbound_id, edge.inbound_id);
355                    self.dedup_cache.edge_filter.set(&edge_key);
356                }
357                logger::info(&format!(
358                    "Preloaded {} edges into bloom filter in {}ms",
359                    count,
360                    start.elapsed().as_millis()
361                ));
362            }
363        }
364        self.dedup_cache.edges_loaded = true;
365        Ok(())
366    }

368    /// Preloads all author vertices into the cache
369    fn preload_authors(
370        &mut self,
371        datastore: &Database<impl Datastore>,
372    ) -> Result<(), Box<dyn std::error::Error>> {
373        let start = Instant::now();
374        logger::info("Preloading authors...");
375
376        if let Ok(person_type) = Identifier::new(crate::db::schema::PERSON_TYPE)
377            && let Ok(name_prop) = Identifier::new("name")
378        {
379            let q = RangeVertexQuery::new().t(person_type);
380
381            if let Ok(q_props) = q.properties()
382                && let Ok(results) = datastore.get(q_props)
383                && let Some(QueryOutputValue::VertexProperties(vps)) = results.first()
384            {
385                let count = vps.len();
386                for vp in vps {
387                    for p in &vp.props {
388                        #[allow(clippy::collapsible_if)]
389                        if p.name == name_prop {
390                            if let Some(name) = p.value.as_str() {
391                                let norm_name = normalize_author(name);
392                                self.author_cache.insert(norm_name, vp.vertex.clone());
393                            }
394                        }
395                    }
396                }
397                logger::info(&format!(
398                    "Preloaded {} authors in {}ms",
399                    count,
400                    start.elapsed().as_millis()
401                ));
402            }
403        }
404        Ok(())
405    }

407    /// Pre-fetches existing edges to warm the cache and limit DB reads
408    fn prefetch_coauthor_edges(
409        &mut self,
410        authors: &[Vertex],
411        datastore: &Database<impl Datastore>,
412    ) -> Result<(), Box<dyn std::error::Error>> {
413        use crate::db::schema::COAUTHORED_WITH_TYPE;
414        use std::collections::HashSet;
415
416        if authors.len() < 2 {
417            return Ok(());
418        }
419
420        let author_ids: Vec<Uuid> = authors.iter().map(|a| a.id).collect();
421        let author_id_set: HashSet<Uuid> = author_ids.iter().cloned().collect();
422        let edge_type = Identifier::new(COAUTHORED_WITH_TYPE)?;
423
424        // Get all outbound edges of type COAUTHORED_WITH from these authors
425        let q = SpecificVertexQuery::new(author_ids)
426            .outbound()?
427            .t(edge_type);
428        let results = datastore.get(q)?;
429
430        let mut relevant_edges = Vec::new();
431
432        if let Some(QueryOutputValue::Edges(edges)) = results.first() {
433            for edge in edges {
434                // Only cache edges that are internal to this group of authors
435                if author_id_set.contains(&edge.inbound_id) {
436                    relevant_edges.push(edge.clone());
437                }
438            }
439        }
440
441        if relevant_edges.is_empty() {
442            return Ok(());
443        }
444
445        // Fetch properties (weight) for these edges
446        let weight_prop = Identifier::new("weight")?;
447        let q_props = SpecificEdgeQuery::new(relevant_edges.clone())
448            .properties()?
449            .name(weight_prop);
450        let prop_results = datastore.get(q_props)?;
451
452        if let Some(QueryOutputValue::EdgeProperties(props_list)) = prop_results.first() {
453            for (i, edge_props) in props_list.iter().enumerate() {
454                if let Some(edge) = relevant_edges.get(i) {
455                    let weight = edge_props
456                        .props
457                        .first()
458                        .and_then(|p| {
459                            p.value.as_u64().or_else(|| {
460                                serde_json::from_value::<String>(p.value.deref().clone())
461                                    .ok()
462                                    .and_then(|s| s.parse::<u64>().ok())
463                            })
464                        })
465                        .unwrap_or(1);
466
467                    self.edge_cache
468                        .insert((edge.outbound_id, edge.inbound_id), weight);
469
470                    // Update bloom filter
471                    let edge_key = format!("{}-{}", edge.outbound_id, edge.inbound_id);
472                    self.dedup_cache.edge_filter.set(&edge_key);
473                }
474            }
475        }
476
477        Ok(())
478    }

481/// Orchestrates the complete scraping process for one or more publication sources.
482///
483/// This function handles:
484/// - Spawning tasks to scrape from specified sources
485/// - Collecting scraped publication records via a thread-safe queue
486/// - Ingesting records into the database
487///
488/// # Arguments
489/// * `start_date` - Start date for scraping
490/// * `end_date` - End date for scraping
491/// * `sources` - List of sources to scrape ("arxiv", "dblp", or "zbmath")
492/// * `datastore` - Mutable reference to the IndraDB datastore
493/// * `config` - Reference to the application configuration
494///
495/// # Returns
496/// `Ok(())` on success, or an error if scraping or ingestion fails
497///
498/// # Errors
499/// - Configuration loading errors
500/// - Scraping errors from individual sources
501/// - Database ingestion errors
502#[coverage(off)]
503pub async fn run_scrape(
504    start_date: DateTime<Utc>,
505    end_date: DateTime<Utc>,
506    sources: Vec<String>,
507    datastore: &mut Database<impl Datastore>,
508    config: &Config,
509) -> Result<(), Box<dyn std::error::Error>> {
510    // Define queue
511    let queue = ThreadSafeQueue::new(QueueConfig::default());
512
513    logger::info(&format!(
514        "Starting scraping for range {} to {}",
515        start_date, end_date
516    ));
517
518    // Define vector of tasks to process
519    let mut tasks = vec![];
520
521    for src in &sources {
522        logger::info(&format!("Spawning scraper task for {}", src));
523        // Create producer task for each source
524        let producer = queue.create_producer();
525        let src_name = src.clone();
526        let start = start_date;
527        let end = end_date;
528        let scraper_config = config.scrapers.clone();
529
530        let task = tokio::spawn(async move {
531            let scraper: Box<dyn Scraper> = match src_name.as_str() {
532                "arxiv" => Box::new(ArxivScraper::with_config(scraper_config.arxiv)),
533                "dblp" => Box::new(DblpScraper::with_config(scraper_config.dblp)),
534                "zbmath" => Box::new(ZbmathScraper::new()), // Zbmath not updated yet
535                _ => {
536                    logger::error(&format!("Unknown source: {}", src_name));
537                    return;
538                }
539            };
540
541            logger::info(&format!("{} scraper started", src_name));
542            match scraper.scrape_range(start, end, producer).await {
543                Ok(()) => {
544                    logger::info(&format!("{} scraper finished", src_name));
545                }
546                Err(e) => {
547                    logger::error(&format!("Scraping failed for {}: {}", src_name, e));
548                }
549            }
550        });
551
552        tasks.push(task);
553    }
554
555    // Consumer loop
556    let mut ingested_count = 0;
557    let mut context = IngestionContext::new(config.deduplication.bloom_filter_size);
558
559    // Preload authors and edges to speed up ingestion
560    if let Err(e) = context.preload_authors(datastore) {
561        logger::error(&format!("Failed to preload authors: {}", e));
562    }
563    if let Err(e) = context.preload_edge_bloom(datastore) {
564        logger::error(&format!("Failed to preload edge bloom filter: {}", e));
565    }
566
567    let loop_start_time = Instant::now();
568    let mut last_log_time = Instant::now();
569    let mut last_log_count = 0;
570
571    // Batching configuration
572    const BATCH_SIZE: usize = 100;
573    const BATCH_TIMEOUT: StdDuration = StdDuration::from_millis(200);
574    let mut batch: Vec<PublicationRecord> = Vec::with_capacity(BATCH_SIZE);
575    let mut last_batch_flush = Instant::now();
576
577    loop {
578        // Attempt to dequeue
579        let record_opt = queue.dequeue();
580        let is_none = record_opt.is_none();
581
582        if let Some(record) = record_opt {
583            batch.push(record);
584        }
585
586        let should_flush = batch.len() >= BATCH_SIZE
587            || (!batch.is_empty() && last_batch_flush.elapsed() >= BATCH_TIMEOUT)
588            || (is_none && queue.producers_finished() && !batch.is_empty());
589
590        if should_flush {
591            let batch_len = batch.len();
592            if let Err(e) =
593                ingest_batch(std::mem::take(&mut batch), datastore, config, &mut context).await
594            {
595                logger::error(&format!("Failed to ingest batch: {}", e));
596            }
597            ingested_count += batch_len;
598            last_batch_flush = Instant::now();
599
600            if ingested_count - last_log_count >= 100 {
601                let now = Instant::now();
602                let elapsed = now.duration_since(last_log_time).as_secs_f64();
603                let total_elapsed = now.duration_since(loop_start_time).as_secs_f64();
604
605                let current_rate = (ingested_count - last_log_count) as f64 / elapsed;
606                let overall_rate = ingested_count as f64 / total_elapsed;
607
608                logger::info(&format!(
609                    "Ingested {} records (Current: {:.2} rec/s, Overall: {:.2} rec/s)",
610                    ingested_count, current_rate, overall_rate
611                ));
612
613                last_log_time = now;
614                last_log_count = ingested_count;
615            }
616        }
617
618        if is_none {
619            if queue.producers_finished() && batch.is_empty() {
620                break;
621            }
622            // Sleep only if we didn't just flush (to avoid busy loop during batch accumulation if queue is slow)
623            if !should_flush {
624                tokio::time::sleep(tokio::time::Duration::from_millis(10)).await; // Reduced sleep for responsiveness
625            }
626        }
627    }
628
629    // Cleanup
630    for task in tasks {
631        let _ = task.await;
632    }
633
634    logger::info("Scraping orchestration finished for current chunk");
635
636    Ok(())
637}

639/// Flushes the write buffer to the database using bulk operations.
640pub(crate) fn flush_buffer(
641    buffer: &mut Vec<WriteOperation>,
642    datastore: &mut Database<impl Datastore>,
643) -> Result<(), Box<dyn std::error::Error>> {
644    if buffer.is_empty() {
645        return Ok(());
646    }
647
648    let start = Instant::now();
649    let count = buffer.len();
650    let mut bulk_items = Vec::with_capacity(count);
651
652    for op in buffer.drain(..) {
653        match op {
654            WriteOperation::CreateVertex(v) => bulk_items.push(BulkInsertItem::Vertex(v)),
655            WriteOperation::CreateEdge(e) => bulk_items.push(BulkInsertItem::Edge(e)),
656            WriteOperation::SetVertexProperties(id, name, val) => {
657                bulk_items.push(BulkInsertItem::VertexProperty(id, name, val));
658            }
659            WriteOperation::SetEdgeProperties(edge, name, val) => {
660                bulk_items.push(BulkInsertItem::EdgeProperty(edge, name, val));
661            }
662        }
663    }
664
665    datastore.bulk_insert(bulk_items)?;
666
667    logger::debug(&format!(
668        "Flushed {} operations in {}ms",
669        count,
670        start.elapsed().as_millis()
671    ));
672
673    Ok(())
674}

678    records: Vec<PublicationRecord>,
679    datastore: &mut Database<impl Datastore>,
680    config: &Config,
681    context: &mut IngestionContext,
682) -> Result<(), Box<dyn std::error::Error>> {
683    let start = Instant::now();
684    let records_len = records.len();
685    if records.is_empty() {
686        return Ok(());
687    }
688
689    // Filter out duplicates
690    let mut new_records = Vec::with_capacity(records.len());
691    let mut batch_seen_ids = HashSet::new();
692
693    for record in records {
694        // Check if we've already seen this ID in the current batch
695        if batch_seen_ids.contains(&record.id) {
696            continue;
697        }
698
699        if !publication_exists(&record, datastore, config, &mut context.dedup_cache) {
700            new_records.push(record.clone());
701            batch_seen_ids.insert(record.id.clone());
702        }
703    }
704
705    if new_records.is_empty() {
706        return Ok(());
707    }
708
709    // Resolve author vertices
710    let mut batch_authors = HashSet::new();
711    for record in &new_records {
712        for author in &record.authors {
713            batch_authors.insert(author.clone());
714        }
715    }
716
717    // Ensure all authors exist using cache or create new vertices
718    for author in &batch_authors {
719        get_or_create_author_vertex(author, &mut context.write_buffer, &mut context.author_cache)?;
720    }
721
722    // Collect author vertices for each record to avoid re-lookup
723    let mut record_author_vertices: Vec<Vec<Vertex>> = Vec::with_capacity(new_records.len());
724    let mut all_involved_authors: HashSet<Uuid> = HashSet::new();
725
726    // Create publication vertices
727    for record in &new_records {
728        let pub_vertex = add_publication(record, &mut context.write_buffer)?;
729        context.dedup_cache.add(
730            record.year,
731            pub_vertex.id,
732            record.title.clone(),
733            &record.authors,
734            record.id.clone(),
735        );
736
737        let mut authors_for_rec = Vec::new();
738        for author_name in &record.authors {
739            let norm = normalize_author(author_name);
740            if let Some(v) = context.author_cache.get(&norm) {
741                authors_for_rec.push(v.clone());
742                all_involved_authors.insert(v.id);
743                create_authored_edge(v, &pub_vertex, &mut context.write_buffer)?;
744            }
745        }
746        record_author_vertices.push(authors_for_rec);
747    }
748
749    // Prefetch and create co-author edges
750    // Collect unique authors to prefetch edges in bulk
751    let unique_author_vertices: Vec<Vertex> = context
752        .author_cache
753        .values()
754        .filter(|v| all_involved_authors.contains(&v.id))
755        .cloned()
756        .collect();
757
758    #[allow(clippy::collapsible_if)]
759    if !unique_author_vertices.is_empty() {
760        if let Err(e) = context.prefetch_coauthor_edges(&unique_author_vertices, datastore) {
761            logger::error(&format!("Failed to prefetch batch co-author edges: {}", e));
762        }
763    }
764
765    // Create co-author edges
766    for authors in record_author_vertices {
767        for i in 0..authors.len() {
768            for j in (i + 1)..authors.len() {
769                let author1 = &authors[i];
770                let author2 = &authors[j];
771                create_coauthor_edge(
772                    author1,
773                    author2,
774                    datastore,
775                    &mut context.write_buffer,
776                    &mut context.edge_cache,
777                    &context.dedup_cache.edge_filter,
778                )?;
779                create_coauthor_edge(
780                    author2,
781                    author1,
782                    datastore,
783                    &mut context.write_buffer,
784                    &mut context.edge_cache,
785                    &context.dedup_cache.edge_filter,
786                )?;
787            }
788        }
789    }
790
791    flush_buffer(&mut context.write_buffer, datastore)?;
792
793    logger::debug(&format!(
794        "Ingested batch of {} records ({} new) in {}ms",
795        records_len,
796        new_records.len(),
797        start.elapsed().as_millis()
798    ));
799
800    Ok(())
801}

803/// Checks if a publication record already exists in the database.
804///
805/// This performs two checks:
806/// 1. Exact match on `publication_id` (e.g., ArXiv ID)
807/// 2. Fuzzy match based on year, title similarity (Cosine), and author set similarity (Jaccard)
808///
809/// # Arguments
810/// * `record` - The publication record to check
811/// * `datastore` - Reference to the IndraDB datastore
812/// * `config` - Reference to the application configuration
813///
814/// # Returns
815/// `true` if a matching publication exists, `false` otherwise
816pub(crate) fn publication_exists(
817    record: &PublicationRecord,
818    datastore: &Database<impl Datastore>,
819    config: &Config,
820    dedup_cache: &mut DeduplicationCache,
821) -> bool {
822    // Check exact match using cache (fast path)
823    // Note: This relies on populate_year loading all IDs for the relevant year
824    // We skip the global DB check for performance (saved ~70ms per record)
825
826    // Fuzzy match using cache
827    dedup_cache.check_exists_and_cache(record, datastore, config)
828}

830/// Helper to check author similarity for a candidate publication
831///
832/// # Arguments
833/// * `pub_id` - The UUID of the existing publication in the database
834/// * `new_authors` - List of authors from the new candidate record
835/// * `datastore` - Reference to the IndraDB datastore
836/// * `threshold` - Jaccard similarity threshold (0.0 to 1.0)
837///
838/// # Returns
839/// `true` if the author sets are similar enough, `false` otherwise
840fn check_authors_similarity(
841    pub_id: &uuid::Uuid,
842    new_authors: &[String],
843    datastore: &Database<impl Datastore>,
844    threshold: f64,
845) -> bool {
846    // Fetch authors of the existing publication
847    // Query: SpecificVertex(pub) -> Inbound (AUTHORED) -> Outbound Vertex (Author) -> Properties
848    let q = SpecificVertexQuery::single(*pub_id);
849    let q_edges = match q.inbound() {
850        Ok(q) => q,
851        Err(_) => return false,
852    };
853    let q_authors = match q_edges.outbound() {
854        Ok(q) => q,
855        Err(_) => return false,
856    };
857
858    let q_props = match q_authors.properties() {
859        Ok(q) => q,
860        Err(_) => return false,
861    };
862
863    let results = match datastore.get(q_props) {
864        Ok(r) => r,
865        Err(_) => return false,
866    };
867
868    let mut db_authors = HashSet::new();
869    if let Some(QueryOutputValue::VertexProperties(vps)) = results.first() {
870        let name_prop = match Identifier::new("name") {
871            Ok(n) => n,
872            Err(_) => return false,
873        };
874
875        for vp in vps {
876            if let Some(val) = vp.props.iter().find(|p| p.name == name_prop)
877                && let Some(name) = val.value.as_str()
878            {
879                db_authors.insert(name.to_lowercase());
880            }
881        }
882    }
883
884    if db_authors.is_empty() && new_authors.is_empty() {
885        return true;
886    }
887    if db_authors.is_empty() || new_authors.is_empty() {
888        return false;
889    }
890
891    let new_authors_set: HashSet<String> = new_authors.iter().map(|s| s.to_lowercase()).collect();
892
893    let intersection = db_authors.intersection(&new_authors_set).count();
894    let union = db_authors.union(&new_authors_set).count();
895
896    let jaccard = if union == 0 {
897        0.0
898    } else {
899        intersection as f64 / union as f64
900    };
901
902    jaccard >= threshold
903}

905/// Adds a publication record to the database.
906///
907/// Creates a new `Publication` vertex and sets its properties.
908///
909/// # Arguments
910/// * `record` - The publication record to add
911/// * `datastore` - Mutable reference to the IndraDB datastore
912///
913/// # Returns
914/// The created Publication `Vertex`
915///
916/// # Errors
917/// Returns an error if database operations fail
918pub(crate) fn add_publication(
919    record: &PublicationRecord,
920    write_buffer: &mut Vec<WriteOperation>,
921) -> Result<Vertex, Box<dyn std::error::Error>> {
922    use crate::db::schema::PUBLICATION_TYPE;
923
924    let vertex_type = Identifier::new(PUBLICATION_TYPE)?;
925    let vertex = Vertex::new(vertex_type);
926
927    write_buffer.push(WriteOperation::CreateVertex(vertex.clone()));
928
929    let properties = vec![
930        ("title", json!(record.title)),
931        ("year", json!(record.year.to_string())),
932        ("venue", json!(record.venue.clone().unwrap_or_default())),
933        ("publication_id", json!(record.id)),
934    ];
935
936    for (name, value) in properties {
937        let prop_name = Identifier::new(name)?;
938        let json_val = Json::new(value);
939        write_buffer.push(WriteOperation::SetVertexProperties(
940            vertex.id, prop_name, json_val,
941        ));
942    }
943
944    Ok(vertex)
945}

947/// Retrieves or creates an author vertex in the database.
948///
949/// # Arguments
950/// * `author_name` - The name of the author
951/// * `datastore` - Mutable reference to the IndraDB datastore
952///
953/// # Returns
954/// The existing or newly created Author `Vertex`
955///
956/// # Errors
957/// Returns an error if database operations fail
958pub(crate) fn get_or_create_author_vertex(
959    author_name: &str,
960    write_buffer: &mut Vec<WriteOperation>,
961    author_cache: &mut HashMap<String, Vertex>,
962) -> Result<Vertex, Box<dyn std::error::Error>> {
963    use crate::db::schema::PERSON_TYPE;
964
965    let norm_name = normalize_author(author_name);
966    if let Some(vertex) = author_cache.get(&norm_name) {
967        return Ok(vertex.clone());
968    }
969
970    let person_type = Identifier::new(PERSON_TYPE)?;
971
972    // TRUST THE CACHE:
973    // Since we preload ALL authors at startup, if it's not in the cache,
974    // it definitely doesn't exist in the DB. Skip the expensive Read check.
975
976    // Create new vertex
977    let vertex = Vertex::new(person_type);
978    write_buffer.push(WriteOperation::CreateVertex(vertex.clone()));
979
980    // Set properties
981    let properties = vec![
982        ("name", json!(author_name)),
983        ("erdos_number", json!("None")),
984        ("is_erdos", json!("false")),
985        ("aliases", json!("[]")),
986        ("updated_at", json!(Utc::now().timestamp().to_string())),
987    ];
988
989    for (name, value) in properties {
990        let prop_name = Identifier::new(name)?;
991        let json_val = Json::new(value);
992        write_buffer.push(WriteOperation::SetVertexProperties(
993            vertex.id, prop_name, json_val,
994        ));
995    }
996
997    author_cache.insert(norm_name, vertex.clone());
998    Ok(vertex)
999}

1028/// Creates or updates a COAUTHORED_WITH edge between two authors.
1029///
1030/// If the edge already exists, its weight is incremented. Otherwise, a new edge
1031/// is created with weight 1.
1032///
1033/// # Arguments
1034/// * `author1` - The first Author vertex
1035/// * `author2` - The second Author vertex
1036/// * `datastore` - Mutable reference to the IndraDB datastore
1037///
1038/// # Returns
1039/// `Ok(())` on success
1040///
1041/// # Errors
1042/// Returns an error if database operations fail
1043pub(crate) fn create_coauthor_edge(
1044    author1: &Vertex,
1045    author2: &Vertex,
1046    datastore: &mut Database<impl Datastore>,
1047    write_buffer: &mut Vec<WriteOperation>,
1048    edge_cache: &mut HashMap<(Uuid, Uuid), u64>,
1049    edge_bloom: &Bloom<String>,
1050) -> Result<(), Box<dyn std::error::Error>> {
1051    use crate::db::schema::COAUTHORED_WITH_TYPE;
1052
1053    let edge_type = Identifier::new(COAUTHORED_WITH_TYPE)?;
1054    let edge = Edge::new(author1.id, edge_type, author2.id);
1055    let weight_prop = Identifier::new("weight")?;
1056    let key = (author1.id, author2.id);
1057
1058    // Check cache first
1059    if let Some(&weight) = edge_cache.get(&key) {
1060        let new_weight = weight + 1;
1061        write_buffer.push(WriteOperation::SetEdgeProperties(
1062            edge,
1063            weight_prop,
1064            Json::new(json!(new_weight)),
1065        ));
1066        edge_cache.insert(key, new_weight);
1067        return Ok(());
1068    }
1069
1070    // Check bloom filter before DB query
1071    let edge_key = format!("{}-{}", author1.id, author2.id);
1072    let maybe_exists = edge_bloom.check(&edge_key);
1073
1074    if !maybe_exists {
1075        // Definitely doesn't exist, create new edge
1076        write_buffer.push(WriteOperation::CreateEdge(edge.clone()));
1077        write_buffer.push(WriteOperation::SetEdgeProperties(
1078            edge,
1079            weight_prop,
1080            Json::new(json!(1u64)),
1081        ));
1082        edge_cache.insert(key, 1);
1083        return Ok(());
1084    }
1085
1086    // Might exist, check DB
1087    let q = SpecificEdgeQuery::single(edge.clone());
1088    let results = datastore.get(q)?;
1089
1090    if let Some(QueryOutputValue::Edges(edges)) = results.first()
1091        && !edges.is_empty()
1092    {
1093        // Edge exists, increment weight
1094        // Get current weight
1095        let q = SpecificEdgeQuery::single(edge.clone())
1096            .properties()?
1097            .name(weight_prop);
1098        let props_results = datastore.get(q)?;
1099
1100        let current_weight = props_results
1101            .first()
1102            .and_then(|res| match res {
1103                QueryOutputValue::EdgeProperties(props) => props.first(),
1104                _ => None,
1105            })
1106            .and_then(|prop| prop.props.first())
1107            .and_then(|p| {
1108                p.value.as_u64().or_else(|| {
1109                    serde_json::from_value::<String>(p.value.deref().clone())
1110                        .ok()
1111                        .and_then(|s| s.parse::<u64>().ok())
1112                })
1113            })
1114            .unwrap_or(1);
1115
1116        let new_weight = current_weight + 1;
1117        write_buffer.push(WriteOperation::SetEdgeProperties(
1118            edge,
1119            weight_prop,
1120            Json::new(json!(new_weight)),
1121        ));
1122
1123        edge_cache.insert(key, new_weight);
1124        return Ok(());
1125    }
1126
1127    // Create new edge
1128    write_buffer.push(WriteOperation::CreateEdge(edge.clone()));
1129    write_buffer.push(WriteOperation::SetEdgeProperties(
1130        edge.clone(),
1131        weight_prop,
1132        Json::new(json!(1u64)),
1133    ));
1134
1135    edge_cache.insert(key, 1);
1136
1137    Ok(())
1138}
