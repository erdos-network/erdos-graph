1//! Orchestrates the scraping process across multiple publication sources.
2//!
3//! This module coordinates scraping from ArXiv, DBLP, and zbMATH, managing
4//! checkpointing, chunking large date ranges, and ingesting results into the database.
5
6use crate::db::ingestion::{get_checkpoint, ingest_publication, set_checkpoint};
7use crate::scrapers::{arxiv, dblp, zbmath};
8use crate::utilities::generate_chunks;
9use chrono::{DateTime, Duration, Utc};
10use indradb::{Database, Datastore};
11
12/// Orchestrates the complete scraping process for one or more publication sources.
13///
14/// This function handles:
15/// - Mode selection (initial full scrape vs. incremental weekly updates)
16/// - Checkpoint restoration to resume from last successful scrape
17/// - Date range chunking to avoid memory issues with large datasets
18/// - Sequential processing of chunks with checkpoint updates
19///
20/// # Arguments
21/// * `mode` - Either "initial" (scrape last 10 years) or "weekly" (incremental updates)
22/// * `start_date` - Optional override for the start date (defaults based on mode)
23/// * `end_date` - Optional override for the end date (defaults to now)
24/// * `chunk_size` - Duration of each scraping chunk (e.g., 7 days)
25/// * `source` - Optional specific source to scrape ("arxiv", "dblp", or "zbmath"), or None for all
26/// * `datastore` - Mutable reference to the IndraDB datastore
27///
28/// # Returns
29/// `Ok(())` on success, or an error if any chunk fails
30///
31/// # Errors
32/// - Invalid mode string
33/// - Checkpoint file I/O errors
34/// - Scraping errors from individual sources
35/// - Database ingestion errors
36#[coverage(off)]
37pub fn run_scrape(
38    mode: &str,
39    start_date: Option<DateTime<Utc>>,
40    end_date: Option<DateTime<Utc>>,
41    chunk_size: Duration,
42    source: Option<&str>,
43    datastore: &mut Database<impl Datastore>,
44) -> Result<(), Box<dyn std::error::Error>> {
45    // Determine which sources to scrape (specific source or all)
46    let sources = source
47        .map(|s| vec![s])
48        .unwrap_or_else(|| vec!["arxiv", "dblp", "zbmath"]);
49
50    for src in sources {
51        let last_checkpoint = get_checkpoint(src)?;
52
53        // Determine effective date range based on mode
54        let effective_start = match mode {
55            "initial" => start_date.unwrap_or(Utc::now() - Duration::days(365 * 10)),
56            "weekly" => last_checkpoint.unwrap_or(Utc::now() - Duration::days(7)),
57            _ => return Err("Invalid mode".into()),
58        };
59        let effective_end = end_date.unwrap_or(Utc::now());
60
61        // Break the date range into manageable chunks
62        let chunks = generate_chunks(effective_start, effective_end, chunk_size);
63
64        // Process each chunk and update checkpoint after success
65        for (chunk_start, chunk_end) in chunks {
66            run_chunk(src, chunk_start, chunk_end, datastore)?;
67            set_checkpoint(src, chunk_end)?;
68        }
69    }
70    Ok(())
71}

73/// Processes a single date range chunk for a specific publication source.
74///
75/// This function:
76/// - Calls the appropriate scraper based on source name
77/// - Retrieves all publication records in the date range
78/// - Ingests each record into the database via `ingest_publication`
79///
80/// # Arguments
81/// * `source` - The source identifier ("arxiv", "dblp", or "zbmath")
82/// * `start_date` - Beginning of the date range (inclusive)
83/// * `end_date` - End of the date range (exclusive)
84/// * `datastore` - Mutable reference to the IndraDB datastore
85///
86/// # Returns
87/// `Ok(())` on success, or an error if scraping or ingestion fails
88///
89/// # Errors
90/// - Unknown source name
91/// - Scraping errors (network, parsing, API limits)
92/// - Database ingestion errors
93#[coverage(off)]
94pub fn run_chunk(
95    source: &str,
96    start_date: DateTime<Utc>,
97    end_date: DateTime<Utc>,
98    datastore: &mut Database<impl Datastore>,
99) -> Result<(), Box<dyn std::error::Error>> {
100    // Dispatch to the appropriate scraper
101    let records = match source {
102        "arxiv" => arxiv::scrape_range(start_date, end_date)?,
103        "dblp" => dblp::scrape_range(start_date, end_date)?,
104        "zbmath" => zbmath::scrape_range(start_date, end_date)?,
105        _ => return Err("Unknown source".into()),
106    };
107
108    // Ingest all records into the database
109    for record in records {
110        ingest_publication(datastore, record)?;
111    }
112    Ok(())
113}
