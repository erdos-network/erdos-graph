21/// Orchestrates the complete scraping process for one or more publication sources.
22///
23/// This function handles:
24/// - Spawning tasks to scrape from specified sources
25/// - Collecting scraped publication records via a thread-safe queue
26/// - Ingesting records into the database
27///
28/// # Arguments
29/// * `start_date` - Start date for scraping
30/// * `end_date` - End date for scraping
31/// * `sources` - List of sources to scrape ("arxiv", "dblp", or "zbmath")
32/// * `datastore` - Mutable reference to the IndraDB datastore
33/// * `config` - Reference to the application configuration
34///
35/// # Returns
36/// `Ok(())` on success, or an error if scraping or ingestion fails
37///
38/// # Errors
39/// - Configuration loading errors
40/// - Scraping errors from individual sources
41/// - Database ingestion errors
42#[coverage(off)]
43pub async fn run_scrape(
44    start_date: DateTime<Utc>,
45    end_date: DateTime<Utc>,
46    sources: Vec<String>,
47    datastore: &mut Database<impl Datastore>,
48    config: &Config,
49) -> Result<(), Box<dyn std::error::Error>> {
50    // Define queue
51    let queue = ThreadSafeQueue::new(QueueConfig::default());
52
53    // Define vector of tasks to process
54    let mut tasks = vec![];
55
56    for src in &sources {
57        // Create producer task for each source
58        let producer = queue.create_producer();
59        let src_name = src.clone();
60        let start = start_date;
61        let end = end_date;
62
63        let task = tokio::spawn(async move {
64            let scraper: Box<dyn Scraper> = match src_name.as_str() {
65                "arxiv" => Box::new(ArxivScraper::new()),
66                "dblp" => Box::new(DblpScraper::new()),
67                "zbmath" => Box::new(ZbmathScraper::new()),
68                _ => {
69                    eprintln!("Unknown source: {}", src_name);
70                    return;
71                }
72            };
73
74            match scraper.scrape_range(start, end).await {
75                Ok(records) => {
76                    for record in records {
77                        if let Err(e) = producer.submit(record) {
78                            eprintln!("Failed to submit record from {}: {}", src_name, e);
79                        }
80                    }
81                }
82                Err(e) => {
83                    eprintln!("Scraping failed for {}: {}", src_name, e);
84                }
85            }
86        });
87
88        tasks.push(task);
89    }
90
91    // Consumer loop
92    loop {
93        if let Some(record) = queue.dequeue() {
94            // Ingest publication into db
95            if let Err(e) = ingest_publication(record, datastore, config).await {
96                eprintln!("Failed to ingest publication: {}", e);
97            }
98        } else {
99            if queue.producers_finished() {
100                break;
101            }
102
103            tokio::time::sleep(tokio::time::Duration::from_millis(
104                config.polling_interval_ms,
105            ))
106            .await;
107        }
108    }
109
110    // Cleanup
111    for task in tasks {
112        let _ = task.await;
113    }
114
115    Ok(())
116}

118/// Ingests a single publication record into the database.
119async fn ingest_publication(
120    record: PublicationRecord,
121    datastore: &mut Database<impl Datastore>,
122    config: &Config,
123) -> Result<(), Box<dyn std::error::Error>> {
124    // Check if record already exists in the database
125    if publication_exists(&record, datastore, config) {
126        return Ok(());
127    }
128
129    // Add publication vertex to database
130    let pub_vertex = add_publication(&record, datastore)?;
131
132    // Add new author vertices and create AUTHORED edges
133    for author in &record.authors {
134        let author_vertex = get_or_create_author_vertex(author, datastore)?;
135        create_authored_edge(&author_vertex, &pub_vertex, datastore)?;
136    }
137
138    // Create COAUTHORED_WITH edges between authors
139    for i in 0..record.authors.len() {
140        for j in (i + 1)..record.authors.len() {
141            // Retrieve author vertices again since we need their IDs
142            // Optimally we'd cache these but for now we look them up
143            let author1 = get_or_create_author_vertex(&record.authors[i], datastore)?;
144            let author2 = get_or_create_author_vertex(&record.authors[j], datastore)?;
145
146            // Create COAUTHORED_WITH edge between record.authors[i] and record.authors[j]
147            create_coauthor_edge(&author1, &author2, datastore)?;
148            // Also create reverse edge for undirected graph simulation
149            create_coauthor_edge(&author2, &author1, datastore)?;
150        }
151    }
152
153    Ok(())
154}

156/// Checks if a publication record already exists in the database.
157///
158/// This performs two checks:
159/// 1. Exact match on `publication_id` (e.g., ArXiv ID)
160/// 2. Fuzzy match based on year, title similarity (Cosine), and author set similarity (Jaccard)
161///
162/// # Arguments
163/// * `record` - The publication record to check
164/// * `datastore` - Reference to the IndraDB datastore
165/// * `config` - Reference to the application configuration
166///
167/// # Returns
168/// `true` if a matching publication exists, `false` otherwise
169pub(crate) fn publication_exists(
170    record: &PublicationRecord,
171    datastore: &Database<impl Datastore>,
172    config: &Config,
173) -> bool {
174    // Check exact match on publication_id
175    if let Ok(id_prop) = Identifier::new("publication_id") {
176        let id_val = Json::new(json!(record.id));
177        let type_prop = match Identifier::new(crate::db::schema::PUBLICATION_TYPE) {
178            Ok(t) => t,
179            Err(_) => return false,
180        };
181
182        if let Ok(q) = RangeVertexQuery::new()
183            .t(type_prop)
184            .with_property_equal_to(id_prop, id_val)
185            && let Ok(results) = datastore.get(q)
186            && let Some(QueryOutputValue::Vertices(v)) = results.first()
187            && !v.is_empty()
188        {
189            return true;
190        }
191    }
192
193    // Fuzzy match within the same year
194    let year_prop = match Identifier::new("year") {
195        Ok(p) => p,
196        Err(_) => return false,
197    };
198    let year_val = Json::new(json!(record.year.to_string()));
199    let type_prop = match Identifier::new(crate::db::schema::PUBLICATION_TYPE) {
200        Ok(t) => t,
201        Err(_) => return false,
202    };
203
204    // Query for papers in the same year and get their properties
205    let q_year = match RangeVertexQuery::new()
206        .t(type_prop)
207        .with_property_equal_to(year_prop, year_val)
208    {
209        Ok(q) => q,
210        Err(_) => return false,
211    };
212
213    let q_props = match q_year.properties() {
214        Ok(q) => q,
215        Err(_) => return false,
216    };
217
218    let results = match datastore.get(q_props) {
219        Ok(r) => r,
220        Err(_) => return false,
221    };
222
223    // Check each candidate publication for title and author similarity
224    if let Some(QueryOutputValue::VertexProperties(vps)) = results.first() {
225        let title_prop_name = match Identifier::new("title") {
226            Ok(n) => n,
227            Err(_) => return false,
228        };
229
230        for vp in vps {
231            // Check title similarity
232            if let Some(title_val) = vp.props.iter().find(|p| p.name == title_prop_name) {
233                let db_title = match title_val.value.as_str() {
234                    Some(s) => s,
235                    None => continue,
236                };
237
238                let cosine = Cosine::default();
239                let db_title_norm = db_title.to_lowercase();
240                let record_title_norm = record.title.to_lowercase();
241                let similarity = cosine.for_str(&record_title_norm, &db_title_norm).nval();
242
243                if similarity >= config.deduplication.title_similarity_threshold {
244                    // Title matches, check authors
245                    if check_authors_similarity(
246                        &vp.vertex.id,
247                        &record.authors,
248                        datastore,
249                        config.deduplication.author_similarity_threshold,
250                    ) {
251                        return true;
252                    }
253                }
254            }
255        }
256    }
257
258    false
259}

261/// Helper to check author similarity for a candidate publication
262///
263/// # Arguments
264/// * `pub_id` - The UUID of the existing publication in the database
265/// * `new_authors` - List of authors from the new candidate record
266/// * `datastore` - Reference to the IndraDB datastore
267/// * `threshold` - Jaccard similarity threshold (0.0 to 1.0)
268///
269/// # Returns
270/// `true` if the author sets are similar enough, `false` otherwise
271fn check_authors_similarity(
272    pub_id: &uuid::Uuid,
273    new_authors: &[String],
274    datastore: &Database<impl Datastore>,
275    threshold: f64,
276) -> bool {
277    // Fetch authors of the existing publication
278    // Query: SpecificVertex(pub) -> Inbound (AUTHORED) -> Outbound Vertex (Author) -> Properties
279    let q = SpecificVertexQuery::single(*pub_id);
280    let q_edges = match q.inbound() {
281        Ok(q) => q,
282        Err(_) => return false,
283    };
284    let q_authors = match q_edges.outbound() {
285        Ok(q) => q,
286        Err(_) => return false,
287    };
288
289    let q_props = match q_authors.properties() {
290        Ok(q) => q,
291        Err(_) => return false,
292    };
293
294    let results = match datastore.get(q_props) {
295        Ok(r) => r,
296        Err(_) => return false,
297    };
298
299    let mut db_authors = HashSet::new();
300    if let Some(QueryOutputValue::VertexProperties(vps)) = results.first() {
301        let name_prop = match Identifier::new("name") {
302            Ok(n) => n,
303            Err(_) => return false,
304        };
305
306        for vp in vps {
307            if let Some(val) = vp.props.iter().find(|p| p.name == name_prop)
308                && let Some(name) = val.value.as_str()
309            {
310                db_authors.insert(name.to_lowercase());
311            }
312        }
313    }
314
315    if db_authors.is_empty() && new_authors.is_empty() {
316        return true;
317    }
318    if db_authors.is_empty() || new_authors.is_empty() {
319        return false;
320    }
321
322    let new_authors_set: HashSet<String> = new_authors.iter().map(|s| s.to_lowercase()).collect();
323
324    let intersection = db_authors.intersection(&new_authors_set).count();
325    let union = db_authors.union(&new_authors_set).count();
326
327    let jaccard = if union == 0 {
328        0.0
329    } else {
330        intersection as f64 / union as f64
331    };
332
333    jaccard >= threshold
334}

336/// Adds a publication record to the database.
337///
338/// Creates a new `Publication` vertex and sets its properties.
339///
340/// # Arguments
341/// * `record` - The publication record to add
342/// * `datastore` - Mutable reference to the IndraDB datastore
343///
344/// # Returns
345/// The created Publication `Vertex`
346///
347/// # Errors
348/// Returns an error if database operations fail
349pub(crate) fn add_publication(
350    record: &PublicationRecord,
351    datastore: &mut Database<impl Datastore>,
352) -> Result<Vertex, Box<dyn std::error::Error>> {
353    use crate::db::schema::PUBLICATION_TYPE;
354
355    let vertex_type = Identifier::new(PUBLICATION_TYPE)?;
356    let vertex = Vertex::new(vertex_type);
357
358    datastore.create_vertex(&vertex)?;
359
360    let properties = vec![
361        ("title", json!(record.title)),
362        ("year", json!(record.year.to_string())),
363        ("venue", json!(record.venue.clone().unwrap_or_default())),
364        ("publication_id", json!(record.id)),
365    ];
366
367    for (name, value) in properties {
368        let q = SpecificVertexQuery::single(vertex.id);
369        let prop_name = Identifier::new(name)?;
370        let json_val = Json::new(value);
371        datastore.set_properties(q, prop_name, &json_val)?;
372    }
373
374    Ok(vertex)
375}

377/// Retrieves or creates an author vertex in the database.
378///
379/// # Arguments
380/// * `author_name` - The name of the author
381/// * `datastore` - Mutable reference to the IndraDB datastore
382///
383/// # Returns
384/// The existing or newly created Author `Vertex`
385///
386/// # Errors
387/// Returns an error if database operations fail
388pub(crate) fn get_or_create_author_vertex(
389    author_name: &str,
390    datastore: &mut Database<impl Datastore>,
391) -> Result<Vertex, Box<dyn std::error::Error>> {
392    use crate::db::schema::PERSON_TYPE;
393
394    let person_type = Identifier::new(PERSON_TYPE)?;
395
396    // Check if author exists by looking up name property
397    let name_prop = Identifier::new("name")?;
398    let name_val = Json::new(json!(author_name));
399
400    let q = RangeVertexQuery::new()
401        .t(person_type)
402        .with_property_equal_to(name_prop, name_val.clone())?;
403
404    let results = datastore.get(q)?;
405
406    if let Some(QueryOutputValue::Vertices(vertices)) = results.first()
407        && let Some(vertex) = vertices.first()
408    {
409        return Ok(vertex.clone());
410    }
411
412    // If not found, create new vertex
413    let vertex = Vertex::new(person_type);
414    datastore.create_vertex(&vertex)?;
415
416    // Set properties
417    let properties = vec![
418        ("name", json!(author_name)),
419        ("erdos_number", json!("None")),
420        ("is_erdos", json!("false")),
421        ("aliases", json!("[]")),
422        ("updated_at", json!(Utc::now().timestamp().to_string())),
423    ];
424
425    for (name, value) in properties {
426        let q = SpecificVertexQuery::single(vertex.id);
427        let prop_name = Identifier::new(name)?;
428        let json_val = Json::new(value);
429        datastore.set_properties(q, prop_name, &json_val)?;
430    }
431
432    Ok(vertex)
433}

435/// Creates an AUTHORED edge between an author and a publication.
436///
437/// # Arguments
438/// * `author` - The Author vertex
439/// * `publication` - The Publication vertex
440/// * `datastore` - Mutable reference to the IndraDB datastore
441///
442/// # Returns
443/// `Ok(())` on success
444///
445/// # Errors
446/// Returns an error if database operations fail
447pub(crate) fn create_authored_edge(
448    author: &Vertex,
449    publication: &Vertex,
450    datastore: &mut Database<impl Datastore>,
451) -> Result<(), Box<dyn std::error::Error>> {
452    use crate::db::schema::AUTHORED_TYPE;
453
454    let edge_type = Identifier::new(AUTHORED_TYPE)?;
455    let edge = Edge::new(author.id, edge_type, publication.id);
456
457    datastore.create_edge(&edge)?;
458
459    Ok(())
460}

462/// Creates or updates a COAUTHORED_WITH edge between two authors.
463///
464/// If the edge already exists, its weight is incremented. Otherwise, a new edge
465/// is created with weight 1.
466///
467/// # Arguments
468/// * `author1` - The first Author vertex
469/// * `author2` - The second Author vertex
470/// * `datastore` - Mutable reference to the IndraDB datastore
471///
472/// # Returns
473/// `Ok(())` on success
474///
475/// # Errors
476/// Returns an error if database operations fail
477pub(crate) fn create_coauthor_edge(
478    author1: &Vertex,
479    author2: &Vertex,
480    datastore: &mut Database<impl Datastore>,
481) -> Result<(), Box<dyn std::error::Error>> {
482    use crate::db::schema::COAUTHORED_WITH_TYPE;
483
484    let edge_type = Identifier::new(COAUTHORED_WITH_TYPE)?;
485    let edge = Edge::new(author1.id, edge_type, author2.id);
486
487    // Check if edge exists
488    let q = SpecificEdgeQuery::single(edge.clone());
489    let results = datastore.get(q)?;
490
491    if let Some(QueryOutputValue::Edges(edges)) = results.first()
492        && !edges.is_empty()
493    {
494        // Edge exists, increment weight
495        let weight_prop = Identifier::new("weight")?;
496
497        // Get current weight
498        let q = SpecificEdgeQuery::single(edge.clone())
499            .properties()?
500            .name(weight_prop);
501        let props_results = datastore.get(q)?;
502
503        let current_weight = props_results
504            .first()
505            .and_then(|res| match res {
506                QueryOutputValue::EdgeProperties(props) => props.first(),
507                _ => None,
508            })
509            .and_then(|prop| prop.props.first())
510            .and_then(|p| {
511                p.value.as_u64().or_else(|| {
512                    serde_json::from_value::<String>(p.value.deref().clone())
513                        .ok()
514                        .and_then(|s| s.parse::<u64>().ok())
515                })
516            })
517            .unwrap_or(1);
518
519        let new_weight = current_weight + 1;
520        let q = SpecificEdgeQuery::single(edge.clone());
521        datastore.set_properties(q, weight_prop, &Json::new(json!(new_weight)))?;
522
523        return Ok(());
524    }
525
526    // Create new edge
527    datastore.create_edge(&edge)?;
528
529    // Initialize properties
530    let weight_prop = Identifier::new("weight")?;
531    let q = SpecificEdgeQuery::single(edge);
532    datastore.set_properties(q, weight_prop, &Json::new(json!(1u64)))?;
533
534    Ok(())
