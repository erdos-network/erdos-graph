1//! Orchestrates the scraping process across multiple publication sources.
2//!
3//! This module coordinates scraping from ArXiv, DBLP, and zbMATH, managing
4//! checkpointing, chunking large date ranges, and ingesting results into the database.
5
6use crate::config::load_config;
7use crate::db::ingestion::{PublicationRecord, get_checkpoint, ingest_publication, set_checkpoint};
8use crate::thread_safe_queue::{QueueConfig, ThreadSafeQueue};
9use chrono::{DateTime, Duration, Utc};
10use indradb::{Database, Datastore};
11use std::collections::HashMap;
12use std::process::Stdio;
13use std::sync::Arc;
14use tokio::io::{AsyncBufReadExt, BufReader};
15use tokio::process::Command;
16use tokio::sync::Mutex;
17use tokio::time::{self, Instant};
18
19/// Orchestrates the complete scraping process for one or more publication sources.
20///
21/// This function handles:
22/// - Mode selection (initial full scrape vs. incremental weekly updates)
23/// - Checkpoint restoration to resume from last successful scrape
24/// - Date range chunking to avoid memory issues with large datasets
25/// - Sequential processing of chunks with checkpoint updates
26///
27/// # Arguments
28/// * `mode` - Either "initial" (scrape last 10 years) or "weekly" (incremental updates)
29/// * `start_date` - Optional override for the start date (defaults based on mode)
30/// * `end_date` - Optional override for the end date (defaults to now)
31/// * `chunk_size` - Duration of each scraping chunk (e.g., 7 days)
32/// * `source` - Optional specific source to scrape ("arxiv", "dblp", or "zbmath"), or None for all
33/// * `datastore` - Mutable reference to the IndraDB datastore
34///
35/// # Returns
36/// `Ok(())` on success, or an error if any chunk fails
37///
38/// # Errors
39/// - Invalid mode string
40/// - Checkpoint file I/O errors
41/// - Scraping errors from individual sources
42/// - Database ingestion errors
43#[coverage(off)]
44pub async fn run_scrape(
45    mode: &str,
46    start_date: Option<DateTime<Utc>>,
47    end_date: Option<DateTime<Utc>>,
48    chunk_size: Duration,
49    source: Option<&str>,
50    datastore: &mut Database<impl Datastore>,
51) -> Result<(), Box<dyn std::error::Error>> {
52    // Load configuration
53    let config = load_config()?;
54
55    // Determine which sources to scrape
56    let sources: Vec<String> = if let Some(src) = source {
57        vec![src.to_string()]
58    } else {
59        config.scrapers.enabled.clone()
60    };
61
62    // Define queue
63    let queue = ThreadSafeQueue::new(QueueConfig::default());
64
65    // Define heartbeat tracking
66    let heartbeat_times = Arc::new(Mutex::new(HashMap::new()));
67    let heartbeat_timeout = time::Duration::from_secs(config.heartbeat_timeout_s);
68
69    // Define vector of sources to process
70    let mut child_handles = vec![];
71
72    for src in &sources {
73        let last_checkpoint = get_checkpoint(src.as_str())?;
74
75        // Determine effective date range based on mode
76        let effective_start = match mode {
77            "initial" => start_date.unwrap_or(Utc::now() - Duration::days(365 * 10)),
78            "weekly" => last_checkpoint.unwrap_or(Utc::now() - Duration::days(7)),
79            _ => return Err("Invalid mode".into()),
80        };
81        let effective_end = end_date.unwrap_or(Utc::now());
82
83        // Spawn child processes
84        let exe = std::env::current_exe()?;
85        let mut cmd = Command::new(exe);
86
87        cmd.arg("scrape-child")
88            .arg("--source")
89            .arg(src)
90            .arg("--start")
91            .arg(effective_start.to_rfc3339())
92            .arg("--end")
93            .arg(effective_end.to_rfc3339())
94            .arg("--chunk-size")
95            .arg(chunk_size.num_days().to_string())
96            .stdout(Stdio::piped())
97            .stderr(Stdio::piped());
98
99        let mut child = cmd.spawn()?;
100        let stdout = child.stdout.take().ok_or("Failed to capture stdout")?;
101        let stderr = child.stderr.take().ok_or("Failed to capture stderr")?;
102
103        // Init heartbeat time
104        {
105            let mut hb_times = heartbeat_times.lock().await;
106            hb_times.insert(src.clone(), Instant::now());
107        }
108
109        // Create producer task for each child process
110        let producer = queue.create_producer();
111        let src_clone = src.clone();
112
113        let handle = tokio::spawn(async move {
114            let mut reader = BufReader::new(stdout).lines();
115
116            while let Ok(Some(line)) = reader.next_line().await {
117                if let Ok(record) = serde_json::from_str::<PublicationRecord>(&line) {
118                    let _ = producer.submit(record);
119                } else {
120                    eprintln!("Failed to parse line from {}: {}", src_clone, line);
121                }
122            }
123        });
124
125        // Create hearbeat reader task
126        let hb_times_clone = heartbeat_times.clone();
127        let src_clone2 = src.clone();
128        let heartbeat_handle = tokio::spawn(async move {
129            let mut reader = BufReader::new(stderr).lines();
130
131            while let Ok(Some(line)) = reader.next_line().await {
132                if line.contains("HEARTBEAT") {
133                    let mut hb_times = hb_times_clone.lock().await;
134                    hb_times.insert(src_clone2.clone(), Instant::now());
135                }
136            }
137        });
138
139        child_handles.push((child, handle, heartbeat_handle, src.clone(), effective_end));
140    }
141
142    // Consumer loop
143    loop {
144        if let Some(record) = queue.dequeue() {
145            ingest_publication(datastore, record)?;
146        } else {
147            if queue.producers_finished() {
148                break;
149            }
150
151            // Check for heartbeat timeouts
152            let hb_times = heartbeat_times.lock().await;
153            let now = Instant::now();
154            for (src, last_heartbeat) in hb_times.iter() {
155                if now.duration_since(*last_heartbeat) > heartbeat_timeout {
156                    eprintln!(
157                        "WARNING: Source {} has not sent a heartbeat in {:?}",
158                        src, heartbeat_timeout
159                    );
160                }
161            }
162            drop(hb_times);
163
164            tokio::time::sleep(tokio::time::Duration::from_millis(
165                config.polling_interval_ms,
166            ))
167            .await;
168        }
169    }
170
171    // Cleanup
172    for (mut child, handle, heartbeat_handle, src, end_date) in child_handles {
173        let _ = handle.await;
174        let _ = heartbeat_handle.await;
175
176        if child.wait().await?.success() {
177            set_checkpoint(&src, end_date)?;
178        }
179    }
180
181    Ok(())
182}
