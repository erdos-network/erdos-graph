1//! Utility functions for date chunking and publication deduplication.
2//!
3//! This module provides helper functions used throughout the scraping and ingestion process.
4
5use chrono::{DateTime, Duration, Utc};
6
7/// Divides a date range into smaller chunks of a specified size.
8///
9/// This is useful for breaking large scraping jobs into manageable pieces,
10/// allowing for checkpointing and avoiding memory issues with large result sets.
11///
12/// # Arguments
13/// * `start` - The beginning of the overall date range
14/// * `end` - The end of the overall date range
15/// * `chunk_size` - The maximum duration of each chunk
16///
17/// # Returns
18/// A vector of (start, end) date pairs representing sequential chunks.
19/// The last chunk may be smaller than `chunk_size` if the range doesn't divide evenly.
20///
21/// # Example
22/// ```
23/// use chrono::{Duration, Utc};
24/// use erdos_graph::utilities::generate_chunks;
25///
26/// let start = Utc::now();
27/// let end = start + Duration::days(30);
28/// let chunks = generate_chunks(start, end, Duration::days(7));
29/// assert_eq!(chunks.len(), 5); // 30 days / 7 days per chunk = 5 chunks
30/// ```
31pub fn generate_chunks(
32    start: DateTime<Utc>,
33    end: DateTime<Utc>,
34    chunk_size: Duration,
35) -> Vec<(DateTime<Utc>, DateTime<Utc>)> {
36    let mut chunks = Vec::new();
37    let mut current = start;
38    while current < end {
39        let next = (current + chunk_size).min(end);
40        chunks.push((current, next));
41        current = next;
42    }
43    chunks
44}

46/// Generates a SHA-256 hash for a publication record to enable deduplication.
47///
48/// This hash can be used to identify duplicate publications from different sources.
49/// The hash is computed from the publication ID, title, concatenated author names, and year.
50///
51/// # Arguments
52/// * `id` - The source-specific publication ID (e.g., ArXiv ID, DBLP key)
53/// * `title` - The publication title
54/// * `authors` - A slice of author names
55/// * `year` - The publication year
56///
57/// # Returns
58/// A hexadecimal string representation of the SHA-256 hash
59///
60/// # Example
61/// ```
62/// use erdos_graph::utilities::hash_publication;
63///
64/// let hash = hash_publication(
65///     "arxiv:2024.12345",
66///     "Graph Theory Applications",
67///     &["Alice".to_string(), "Bob".to_string()],
68///     2024
69/// );
70/// assert_eq!(hash.len(), 64); // SHA-256 produces 64 hex characters
71/// ```
72pub fn hash_publication(id: &str, title: &str, authors: &[String], year: u32) -> String {
73    use sha2::{Digest, Sha256};
74    let mut hasher = Sha256::new();
75    hasher.update(format!("{}{}{}{}", id, title, authors.join(","), year));
76    format!("{:x}", hasher.finalize())
77}
