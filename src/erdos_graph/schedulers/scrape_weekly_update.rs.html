2//! Weekly scraping scheduler
3//!
4//! This module sets up a cron-based scheduler that runs weekly scrapes for all enabled sources.
5
6use crate::config::Config;
7use crate::db::ingestion::orchestrate_scraping_and_ingestion;
8use helix_db::helix_engine::traversal_core::HelixGraphEngine;
9use std::sync::Arc;
10use tokio_cron_scheduler::{Job, JobScheduler};
11
12/// Starts the weekly scraping scheduler.
13///
14/// This function creates a cron job that runs every week to scrape new publications
15/// from all enabled sources defined in the configuration.
16///
17/// # Arguments
18/// * `config` - Configuration containing enabled sources and ingestion settings
19/// * `datastore` - Arc-wrapped HelixGraphEngine
20///
21/// # Returns
22/// `Ok(JobScheduler)` on success, which must be kept alive for the scheduler to run
23pub async fn start_weekly_scraper(
24    config: Config,
25    datastore: Arc<HelixGraphEngine>,
26) -> Result<JobScheduler, Box<dyn std::error::Error>> {
27    let scheduler = JobScheduler::new().await?;
28
29    let config_clone = config.clone();
30    let sources = config.scrapers.enabled.clone();
31
32    // Create a weekly job (runs every Sunday at 2 AM)
33    let job = Job::new_async("0 0 2 * * Sun", move |_uuid, _lock| {
34        let sources = sources.clone();
35        let datastore = datastore.clone();
36        let config = config_clone.clone();
37
38        Box::pin(async move {
39            println!("Starting weekly scrape for sources: {:?}", sources);
40
41            match orchestrate_scraping_and_ingestion("weekly", sources.clone(), datastore, &config)
42                .await
43            {
44                Ok(_) => println!(
45                    "Weekly scrape completed successfully for sources: {:?}",
46                    sources
47                ),
48                Err(e) => eprintln!("Weekly scrape failed: {}", e),
49            }
50        })
51    })?;
52
53    scheduler.add(job).await?;
54
55    Ok(scheduler)
56}
