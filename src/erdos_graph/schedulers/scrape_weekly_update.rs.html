2//! Weekly scraping scheduler
3//!
4//! This module sets up a cron-based scheduler that runs weekly scrapes for all enabled sources.

6use crate::config::Config;
7use crate::db::ingestion::orchestrate_scraping_and_ingestion;
8use indradb::{Database, Datastore};
9use std::sync::Arc;
10use tokio::sync::Mutex;
11use tokio_cron_scheduler::{Job, JobScheduler};
12
13/// Starts the weekly scraping scheduler.
14///
15/// This function creates a cron job that runs every week to scrape new publications
16/// from all enabled sources defined in the configuration.
17///
18/// # Arguments
19/// * `config` - Configuration containing enabled sources and ingestion settings
20/// * `datastore` - Arc-wrapped Mutex of the IndraDB datastore
21///
22/// # Returns
23/// `Ok(JobScheduler)` on success, which must be kept alive for the scheduler to run
24///
25/// # Example
26/// ```no_run
27/// # use erdos_graph::config::Config;
28/// # use erdos_graph::schedulers::scrape_weekly_update::start_weekly_scraper;
29/// # use indradb::Database;
30/// # use std::sync::Arc;
31/// # use tokio::sync::Mutex;
32/// #
33/// # async fn example<D: indradb::Datastore + Send + 'static>(
34/// #     config: Config,
35/// #     datastore: Arc<Mutex<Database<D>>>,
36/// # ) -> Result<(), Box<dyn std::error::Error>> {
37/// let scheduler = start_weekly_scraper(config, datastore).await?;
38/// scheduler.start().await?;
39/// # Ok(())
40/// # }
41/// ```
42pub async fn start_weekly_scraper<D: Datastore + Send + 'static>(
43    config: Config,
44    datastore: Arc<Mutex<Database<D>>>,
45) -> Result<JobScheduler, Box<dyn std::error::Error>> {
46    let scheduler = JobScheduler::new().await?;
47
48    let config_clone = config.clone();
49    let sources = config.scrapers.enabled.clone();
50
51    // Create a weekly job (runs every Sunday at 2 AM)
52    let job = Job::new_async("0 0 2 * * Sun", move |_uuid, _lock| {
53        let sources = sources.clone();
54        let datastore = datastore.clone();
55        let config = config_clone.clone();
56
57        Box::pin(async move {
58            println!("Starting weekly scrape for sources: {:?}", sources);
59
60            let mut db = datastore.lock().await;
61
62            match orchestrate_scraping_and_ingestion("weekly", sources.clone(), &mut *db, &config)
63                .await
64            {
65                Ok(_) => println!(
66                    "Weekly scrape completed successfully for sources: {:?}",
67                    sources
68                ),
69                Err(e) => eprintln!("Weekly scrape failed: {}", e),
70            }
71        })
72    })?;
73
74    scheduler.add(job).await?;
75
76    Ok(scheduler)
77}
