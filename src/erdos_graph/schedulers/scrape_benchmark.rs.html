2//! Benchmark scraping job
3//!
4//! This module provides a job to benchmark the scraping process for a specified number of weeks.

6use crate::config::Config;
7use crate::db::ingestion::orchestrate_scraping_and_ingestion;
8use crate::logger;
9use helix_db::helix_engine::traversal_core::HelixGraphEngine;
10use std::sync::Arc;
11use std::time::Instant;
12
13/// Runs a benchmark scrape for a specified duration (in weeks).
14///
15/// This job modifies the configuration to scrape back `num_weeks` and times the execution.
16/// It acquires the datastore lock to ensure exclusive access.
17/// It uses a temporary checkpoint directory to ignore existing checkpoints and force
18/// scraping of the requested duration.
19///
20/// # Arguments
21/// * `num_weeks` - Number of weeks to scrape back
22/// * `config` - Base application configuration
23/// * `datastore` - Arc-wrapped HelixGraphEngine
24///
25/// # Returns
26/// `Ok(Duration)` containing the elapsed time, or an error if execution fails
27pub async fn run_benchmark(
28    num_weeks: u64,
29    mut config: Config,
30    datastore: Arc<HelixGraphEngine>,
31) -> Result<std::time::Duration, Box<dyn std::error::Error>> {
32    let sources = config.scrapers.enabled.clone();
33    logger::info(&format!(
34        "Starting benchmark scrape for {} weeks on sources: {:?}",
35        num_weeks, sources
36    ));
37
38    // Use subdirectories of configured paths for benchmark artifacts to ensure isolation and respect gitignore
39    let base_checkpoint_dir = config
40        .ingestion
41        .checkpoint_dir
42        .as_deref()
43        .unwrap_or("checkpoints");
44    let benchmark_checkpoint_dir = std::path::Path::new(base_checkpoint_dir).join("benchmark");
45
46    let base_cache_dir = &config.scrapers.dblp.cache_dir;
47    let benchmark_cache_dir = std::path::Path::new(base_cache_dir).join("benchmark");
48
49    // Clean up existing benchmark directories to ensure a fresh run
50    if benchmark_checkpoint_dir.exists() {
51        std::fs::remove_dir_all(&benchmark_checkpoint_dir)?;
52    }
53    // We also clean the cache to benchmark the full scraping process including network requests
54    if benchmark_cache_dir.exists() {
55        std::fs::remove_dir_all(&benchmark_cache_dir)?;
56    }
57
58    config.ingestion.checkpoint_dir = Some(benchmark_checkpoint_dir.to_string_lossy().to_string());
59    config.scrapers.dblp.cache_dir = benchmark_cache_dir.to_string_lossy().to_string();
60
61    // Override weekly_days in config for the benchmark
62    config.ingestion.weekly_days = num_weeks * 7;
63
64    let start_time = Instant::now();
65
66    match orchestrate_scraping_and_ingestion("weekly", sources.clone(), datastore, &config).await {
67        Ok(_) => {
68            let duration = start_time.elapsed();
69            logger::info(&format!(
70                "Benchmark completed successfully. Elapsed time: {:.2} seconds",
71                duration.as_secs_f64()
72            ));
73            Ok(duration)
74        }
75        Err(e) => {
76            logger::error(&format!("Benchmark failed: {}", e));
77            Err(e)
78        }
79    }
80}
