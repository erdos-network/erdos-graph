2//! Full refresh scraping job
3//!
4//! This module provides a job to perform a full initial scrape for specified sources.
5
6use crate::config::Config;
7use crate::db::ingestion::orchestrate_scraping_and_ingestion;
8use helix_db::helix_engine::traversal_core::HelixGraphEngine;
9use std::sync::Arc;
10
11/// Runs a full refresh scrape for the specified sources.
12///
13/// This function acquires the datastore lock to ensure exclusive access
14/// and runs the scraping pipeline in "initial" mode.
15///
16/// # Arguments
17/// * `sources` - List of sources to scrape (e.g., "arxiv", "dblp")
18/// * `config` - Application configuration
19/// * `datastore` - Arc-wrapped HelixGraphEngine
20///
21/// # Returns
22/// `Ok(())` on success, or an error if scraping fails
23pub async fn run_full_refresh(
24    sources: Vec<String>,
25    config: Config,
26    datastore: Arc<HelixGraphEngine>,
27) -> Result<(), Box<dyn std::error::Error>> {
28    println!("Starting full refresh scrape for sources: {:?}", sources);
29
30    // Use XML mode for DBLP in full refresh (faster for large date ranges)
31    let mut source_modes = std::collections::HashMap::new();
32    if sources.contains(&"dblp".to_string()) {
33        source_modes.insert("dblp".to_string(), "xml".to_string());
34    }
35
36    match orchestrate_scraping_and_ingestion(
37        "initial",
38        sources.clone(),
39        Some(source_modes),
40        datastore,
41        &config,
42    )
43    .await
44    {
45        Ok(_) => {
46            println!(
47                "Full refresh scrape completed successfully for sources: {:?}",
48                sources
49            );
50            Ok(())
51        }
52        Err(e) => {
53            eprintln!("Full refresh scrape failed: {}", e);
54            Err(e)
55        }
56    }
57}
